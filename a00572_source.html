201 MatrixX4f FsSurfaceTreeItem::createCurvatureVertColor(const VectorXf& curvature,
202  const QColor& colSulci,
203  const QColor& colGyri)
204 {
205  MatrixX4f colors(curvature.rows(), 4);
206 
207  for(int i = 0; i < colors.rows(); ++i) {
208  //Color (this is the default color and will be used until the updateVertColor function was called)
209  if(curvature(i) >= 0) {
210  colors(i,0) = colSulci.redF();
211  colors(i,1) = colSulci.greenF();
212  colors(i,2) = colSulci.blueF();
213  colors(i,3) = colSulci.alphaF();
214  } else {
215  colors(i,0) = colGyri.redF();
216  colors(i,1) = colGyri.greenF();
217  colors(i,2) = colGyri.blueF();
218  colors(i,3) = colGyri.alphaF();
219  }
220  }
221 
222  return colors;
223 }
