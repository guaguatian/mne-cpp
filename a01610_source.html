406 VectorXd AdaptiveMp::calculate_atom(qint32 sample_count, qreal scale, qint32 translation, qreal modulation, qint32 channel, MatrixXd residuum, ReturnValue return_value = RETURNATOM, bool fix_phase = false)
407 {
408  GaborAtom *gabor_Atom = new GaborAtom();
409  qreal phase = 0;
410  //create complex Gaboratom
411  VectorXcd complex_gabor_atom = gabor_Atom->create_complex(sample_count, scale, translation, modulation);
412 
413  //calculate Inner Product: preparation to find the parameter phase
414  std::complex<double> inner_product(0, 0);
415 
416  if(fix_phase == false)
417  {
418  for(qint32 i = 0; i < sample_count; i++)
419  inner_product += residuum(i, channel) * conj(complex_gabor_atom[i]);
420  }
421  else
422  {
423  for(qint32 chn = 0; chn < residuum.cols(); chn++)
424  {
425  for(qint32 i = 0; i < sample_count; i++)
426  inner_product += residuum(i, chn) * conj(complex_gabor_atom[i]);
427  }
428  if(residuum.cols() != 0)
429  inner_product /= residuum.cols();
430  }
431 
432  //calculate phase to create realGaborAtoms
433  phase = std::arg(inner_product);
434  if (phase < 0) phase = 2 * PI - phase;
435  VectorXd real_gabor_atom = gabor_Atom->create_real(sample_count, scale, translation, modulation, phase);
436 
437  delete gabor_Atom;
438 
439  switch(return_value)
440  {
441  case RETURNPARAMETERS:
442  {
443 
444  qreal scalar_product = 0;
445 
446  for(qint32 i = 0; i < sample_count; i++)
447  scalar_product += real_gabor_atom[i] * residuum(i, channel);
448 
449  VectorXd atom_parameters = VectorXd::Zero(5);
450 
451  atom_parameters[0] = scale;
452  atom_parameters[1] = translation;
453  atom_parameters[2] = modulation;
454  atom_parameters[3] = phase;
455  atom_parameters[4] = scalar_product;
456 
457 
458  return atom_parameters;
459  }
460 
461  case RETURNATOM: {return real_gabor_atom;} //returns normalized realGaborAtom
462

464  }

469 void AdaptiveMp::simplex_maximisation(qint32 simplex_it, qreal simplex_reflection, qreal simplex_expansion, qreal simplex_contraction, qreal simplex_full_contraction,

472  //Maximisation Simplex Algorithm implemented by Botao Jia, adapted to the MP Algorithm by Martin Henfling. Copyright (C) 2010 Botao Jia
473  //ToDo: change to clean use of EIGEN, @present its mixed with Namespace std and <vector>
474  std::vector<double> init;
475 
476  init.push_back(gabor_Atom->scale);
477  init.push_back(gabor_Atom->translation);
478  init.push_back(gabor_Atom->modulation);
479 
480  double tol = 1E8 * std::numeric_limits<double>::epsilon();
481  std::vector<std::vector<double> > x = std::vector<std::vector<double> >();
482  qint32 iterations = simplex_it;
483  qint32 N = qint32 (init.size()); //space dimension
484 
485  VectorXd atom_fxc_params = VectorXd::Zero(5); //initialisation for contraction coefficients
486 
487  qreal a = simplex_reflection, b = simplex_expansion, g = simplex_contraction, h = simplex_full_contraction;
488  //coefficients default a = 1, b = 0.2, g = 0.5, h = 0.5
489  //a: reflection -> xr step away from worst siplex found
490  //b: expansion -> xe if better with a so go in this direction with b
491  //g: contraction -> xc calc new worst point an bring closer to middle of simplex
492  //h: full contraction to x1
493  std::vector<double> xcentroid_old(N,0); //simplex center * (N+1)
494  std::vector<double> xcentroid_new(N,0); //simplex center * (N+1)

496  qint32 x1 = 0, xn = 0, xnp1 = 0; //x1: f(x1) = min { f(x1), f(x2)...f(x_{n+1} }
497  //xnp1: f(xnp1) = max { f(x1), f(x2)...f(x_{n+1} }
498  //xn: f(xn)<f(xnp1) && f(xn)> all other f(x_i)
499  qint32 cnt = 0; //iteration step number
500 
501  if(x.size()== 0) //if no initial simplex is specified
502  {
503  //construct the trial simplex
504  //based upon the initial guess parameters
505  std::vector<double> del( init );
506  std::transform(del.begin(), del.end(), del.begin(),
507  std::bind2nd( std::divides<double>() , 20) );//'20' is picked
508  //assuming initial trail close to true
509

511  {
512  std::vector<double> tmp( init );
513  tmp[i] += del[i];

521  }//constructing the simplex finished
522

525  {

527  {

536  //create targetfunction of realGaborAtom and Residuum
537  double target = 0;
538  for(qint32 k = 0; k < atom_fx.rows(); k++)
539  {
540  target -=atom_fx[k]*residuum(k, chn); //ToDo: old residuum(k,0)
541  }

544  }

546  x1=0; xn=0; xnp1=0;//find index of max, second max, min of vf.
547 
548  for(quint32 i=0; i < vf.size(); ++i)
549  {
550  if(vf[i]<vf[x1]) x1 = i;
551  if(vf[i]>vf[xnp1]) xnp1 = i;
552  }
553 
554  xn = x1;
555 
556  for(quint32 i=0; i<vf.size();++i) if(vf[i]<vf[xnp1] && vf[i]>vf[xn]) xn=i;
