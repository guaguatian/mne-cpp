129 void EcdDataTreeItem::plotDipoles(const ECDSet& tECDSet)
130 {
131  //Plot dipoles
132 
133  //create geometry
134  QSharedPointer<Qt3DExtras::QConeGeometry> pDipolGeometry = QSharedPointer<Qt3DExtras::QConeGeometry>::create();
135  pDipolGeometry->setBottomRadius(0.001f);
136  pDipolGeometry->setLength(0.003f);
137  //create instanced renderer
138  GeometryMultiplier *pDipolMesh = new GeometryMultiplier(pDipolGeometry);
139 
140  QVector3D pos, to, from;
141 
142  //The Qt3D default cone orientation and the top of the cone lies in line with the positive y-axis.
143  from = QVector3D(0.0, 1.0, 0.0);
144  double norm;
145 
146  QVector<QColor> vColors;
147  vColors.reserve(tECDSet.size());
148  QVector<QMatrix4x4> vTransforms;
149  vTransforms.reserve(tECDSet.size());
150 
151  for(int i = 0; i < tECDSet.size(); ++i) {
152  pos.setX(tECDSet[i].rd(0));
153  pos.setY(tECDSet[i].rd(1));
154  pos.setZ(tECDSet[i].rd(2));
155 
156  norm = sqrt(pow(tECDSet[i].Q(0),2)+pow(tECDSet[i].Q(1),2)+pow(tECDSet[i].Q(2),2));
157 
158  to.setX(tECDSet[i].Q(0)/norm);
159  to.setY(tECDSet[i].Q(1)/norm);
160  to.setZ(tECDSet[i].Q(2)/norm);
161 
162 // qDebug()<<"EcdDataTreeItem::plotDipoles - from" << from;
163 // qDebug()<<"EcdDataTreeItem::plotDipoles - to" << to;
164 
165  QQuaternion final = QQuaternion::rotationTo(from, to);
166 
167  //Set dipole position and orientation
168  QMatrix4x4 m;
169  m.translate(pos);
170  m.rotate(final);
171  vTransforms.push_back(m);
172 
173  //add random color;
174  vColors.push_back(QColor(QRandomGenerator::global()->bounded(0 , 255),
175  QRandomGenerator::global()->bounded(0 , 255),
176  QRandomGenerator::global()->bounded(0 , 255)));
177  }
178  //Set instance Transform
179  pDipolMesh->setTransforms(vTransforms);
180  //Set instance colors
181  pDipolMesh->setColors(vColors);
182 
183  this->addComponent(pDipolMesh);
184 
185  //Add material
186  GeometryMultiplierMaterial* pMaterial = new GeometryMultiplierMaterial;
187  pMaterial->setAmbient(QColor(0,0,0));
188  pMaterial->setAlpha(1.0f);
189  this->addComponent(pMaterial);
190 }
