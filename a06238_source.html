92 void generateRandomDataSamples(int sample_size, int feature_dim, int num_classes, MatrixXf& X, MatrixXf& Y)
93 {
94  MatrixXi t_Y = MatrixXi::Zero(sample_size, 1);
95  for(int i = 0; i < t_Y.rows(); ++i) {
96  t_Y(i,0) = rand() % num_classes;
97  }
98 
99  std::default_random_engine generator;
100  std::normal_distribution<float> distribution(0.0,1.0);
101 
102  X = MatrixXf::Zero(sample_size, feature_dim);
103  for(int i = 0; i < X.rows(); ++i) {
104  for(int j = 0; j < X.cols(); ++j) {
105  float number = distribution(generator);
106  X(i,j) = (number + 3) * (t_Y(i) + 1);
107  }
108  }
109 
110  Y = MatrixXf::Zero(sample_size, num_classes);
111 
112  for(int i = 0; i < Y.rows(); ++i) {
113  Y(i,t_Y(i)) = 1;
114  }
115 }
