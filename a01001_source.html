183  int *neegp,
184  FiffCoordTransOld* *meg_head_t,
185  fiffId *idp) /* The measurement ID */
186 /*
187  * Read the channel information and split it into three arrays,
188  * one for MEG, one for MEG compensation channels, and one for EEG
189  */
190 {
191  QFile file(name);
192  FiffStream::SPtr stream(new FiffStream(&file));
193 
194 
195  fiffChInfo chs = NULL;
196  int nchan = 0;
197  fiffChInfo meg = NULL;
198  int nmeg = 0;
199  fiffChInfo meg_comp = NULL;
200  int nmeg_comp = 0;
201  fiffChInfo eeg = NULL;
202  int neeg = 0;
203  fiffId id = NULL;
204  QList<FiffDirNode::SPtr> nodes;
205  FiffDirNode::SPtr info;
206  FiffTag::SPtr t_pTag;
207  fiffChInfo this_ch = NULL;
208  FiffCoordTransOld* t = NULL;
209  fiff_int_t kind, pos;
210  int j,k,to_find;
211 
212  if(!stream->open())
213  goto bad;
214 
215  nodes = stream->dirtree()->dir_tree_find(FIFFB_MNE_PARENT_MEAS_FILE);
216 
217  if (nodes.size() == 0) {
218  nodes = stream->dirtree()->dir_tree_find(FIFFB_MEAS_INFO);
219  if (nodes.size() == 0) {
220  qCritical ("Could not find the channel information.");
221  goto bad;
222  }
223  }
224  info = nodes[0];
225  to_find = 0;
226  for (k = 0; k < info->nent(); k++) {
227  kind = info->dir[k]->kind;
228  pos = info->dir[k]->pos;
229  switch (kind) {
230  case FIFF_NCHAN :
231  if (!stream->read_tag(t_pTag,pos))
232  goto bad;
233  nchan = *t_pTag->toInt();
234  chs = MALLOC_41(nchan,fiffChInfoRec);
235  for (j = 0; j < nchan; j++)
236  chs[j].scanNo = -1;
237  to_find = nchan;
238  break;
239 
240  case FIFF_PARENT_BLOCK_ID :
241  if(!stream->read_tag(t_pTag, pos))
242  goto bad;
243  // id = t_pTag->toFiffID();
244  *id = *(fiffId)t_pTag->data();
245  break;
246 
247  case FIFF_COORD_TRANS :
248  if(!stream->read_tag(t_pTag, pos))
249  goto bad;
250  // t = t_pTag->toCoordTrans();
251  t = FiffCoordTransOld::read_helper( t_pTag );
252  if (t->from != FIFFV_COORD_DEVICE || t->to != FIFFV_COORD_HEAD)
253  t = NULL;
254  break;
255 
256  case FIFF_CH_INFO : /* Information about one channel */
257  if(!stream->read_tag(t_pTag, pos))
258  goto bad;
259  // this_ch = t_pTag->toChInfo();
260  this_ch = (fiffChInfo)malloc(sizeof(fiffChInfoRec));
261  *this_ch = *(fiffChInfo)(t_pTag->data());
262  if (this_ch->scanNo <= 0 || this_ch->scanNo > nchan) {
263  printf ("FIFF_CH_INFO : scan # out of range %d (%d)!",this_ch->scanNo,nchan);
264  goto bad;
265  }
266  else
267  chs[this_ch->scanNo-1] = *this_ch;
268  to_find--;
269  break;
270  }
271  }
272  if (to_find != 0) {
273  qCritical("Some of the channel information was missing.");
274  goto bad;
275  }
276  if (t == NULL && meg_head_t != NULL) {
277  /*
278  * Try again in a more general fashion
279  */
280  if ((t = FiffCoordTransOld::mne_read_meas_transform(name)) == NULL) {
281  qCritical("MEG -> head coordinate transformation not found.");
282  goto bad;
283  }
284  }
285  /*
286  * Sort out the channels
287  */
288  for (k = 0; k < nchan; k++)
289  if (chs[k].kind == FIFFV_MEG_CH)
290  nmeg++;
291  else if (chs[k].kind == FIFFV_REF_MEG_CH)
292  nmeg_comp++;
293  else if (chs[k].kind == FIFFV_EEG_CH)
294  neeg++;
295  if (nmeg > 0)
296  meg = MALLOC_41(nmeg,fiffChInfoRec);
297  if (neeg > 0)
298  eeg = MALLOC_41(neeg,fiffChInfoRec);
299  if (nmeg_comp > 0)
300  meg_comp = MALLOC_41(nmeg_comp,fiffChInfoRec);
301  neeg = nmeg = nmeg_comp = 0;
302 
303  for (k = 0; k < nchan; k++)
304  if (chs[k].kind == FIFFV_MEG_CH)
305  meg[nmeg++] = chs[k];
306  else if (chs[k].kind == FIFFV_REF_MEG_CH)
307  meg_comp[nmeg_comp++] = chs[k];
308  else if (chs[k].kind == FIFFV_EEG_CH)
309  eeg[neeg++] = chs[k];
310  // fiff_close(in);
311  stream->close();
312  FREE_41(chs);
313  if (megp) {
314  *megp = meg;
315  *nmegp = nmeg;
316  }
317  else
318  FREE_41(meg);
319  if (meg_compp) {
320  *meg_compp = meg_comp;
321  *nmeg_compp = nmeg_comp;
322  }
323  else
324  FREE_41(meg_comp);
325  if (eegp) {
326  *eegp = eeg;
327  *neegp = neeg;
328  }
329  else
330  FREE_41(eeg);
331  if (idp == NULL) {
332  FREE_41(id);
333  }
334  else
335  *idp = id;
336  if (meg_head_t == NULL) {
337  FREE_41(t);
338  }
339  else
340  *meg_head_t = t;
341 
342  return FIFF_OK;
343 
344 bad : {
345  // fiff_close(in);
346  stream->close();
347  FREE_41(chs);
348  FREE_41(meg);
349  FREE_41(eeg);
350  FREE_41(id);
351  // FREE_41(tag.data);
352  FREE_41(t);
353  return FIFF_FAIL;
354  }
355 }
