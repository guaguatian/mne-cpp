447 void RealTimeEvokedSetWidget::onMakeScreenshot(const QString& imageType)
448 {
449  // Create file name
450  QString sDate = QDate::currentDate().toString("yyyy_MM_dd");
451  QString sTime = QTime::currentTime().toString("hh_mm_ss");
452 
453  if(!QDir("./Screenshots").exists()) {

455  }
456 
457  //Handle the butterfly plot and 2D layout plot differently

459

464  fileName = QString("./Screenshots/%1-%2-LayoutScreenshot.png").arg(sDate).arg(sTime);
