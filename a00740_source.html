299  // Y - red

303

307  vTransforms.push_back(transformMat);
308  vColors.push_back(QColor(0, 0, 255));
309 
310  // Z - green

313  vTransforms.push_back(transformMat);
314  vColors.push_back(QColor(0, 255, 0));
315 
316  //Set transforms and colors
317  pCoordSysMesh->setTransforms(vTransforms);
318  pCoordSysMesh->setColors(vColors);
319 
320  //Add material

322  pCoordSysMaterial->setAmbient(QColor(0,0,0));
