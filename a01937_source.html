411 bool DataWindow::eventFilter(QObject *object, QEvent *event)
412 {
413  //Detect double mouse clicks and move data marker to current mouse position
414  if (object == ui->m_tableView_rawTableView->viewport() && event->type() == QEvent::MouseButtonDblClick) {
415  QMouseEvent* mouseEventCast = static_cast<QMouseEvent*>(event);
416  if(mouseEventCast->button() == Qt::LeftButton) {
417  m_pDataMarker->move(mouseEventCast->localPos().x() + ui->m_tableView_rawTableView->verticalHeader()->width() + ui->m_tableView_rawTableView->x(), m_pDataMarker->y());
418 
419  //Deselect channel which was selected through the double click -> dirty hack
420  //ui->m_tableView_rawTableView->selectionModel()->select(ui->m_tableView_rawTableView->selectionModel()->currentIndex(), QItemSelectionModel::Deselect);
421  }
422  }
423 
424  //Deactivate grabbing gesture when scrollbars or vertical header are selected
425  if ((object == ui->m_tableView_rawTableView->horizontalScrollBar() ||
426  object == ui->m_tableView_rawTableView->verticalScrollBar() ||
427  object == ui->m_tableView_rawTableView->verticalHeader())
428  && event->type() == QEvent::Enter) {
429  QScroller::ungrabGesture(ui->m_tableView_rawTableView);
430  return true;
431  }
432 
433  //Activate grabbing gesture when scrollbars or vertical header are deselected
434  if ((object == ui->m_tableView_rawTableView->horizontalScrollBar() ||
435  object == ui->m_tableView_rawTableView->verticalScrollBar() ||
436  object == ui->m_tableView_rawTableView->verticalHeader())
437  && event->type() == QEvent::Leave) {
438  QScroller::grabGesture(ui->m_tableView_rawTableView, QScroller::LeftMouseButtonGesture);
439  return true;
440  }
441 
442  //Look for swipe gesture in order to scale the channels
443  if (object == ui->m_tableView_rawTableView && event->type() == QEvent::Gesture) {
444  QGestureEvent* gestureEventCast = static_cast<QGestureEvent*>(event);
445  return gestureEvent(static_cast<QGestureEvent*>(gestureEventCast));
446  }
447 
448  return false;
449 }

650 bool DataWindow::pinchTriggered(QPinchGesture *gesture)
651 {
652  QPinchGesture::ChangeFlags changeFlags = gesture->changeFlags();
653  if (changeFlags & QPinchGesture::ScaleFactorChanged) {
654  emit scaleChannels(gesture->scaleFactor());
655  qDebug()<<"ungrab";
656  ui->m_tableView_rawTableView->setSelectionMode(QAbstractItemView::NoSelection);
657  QScroller::ungrabGesture(ui->m_tableView_rawTableView);
658  }
659 
660  if (gesture->state() == Qt::GestureFinished) {
