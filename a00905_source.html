155 bool FiffInfo::make_compensator(fiff_int_t from, fiff_int_t to, FiffCtfComp& ctf_comp, bool exclude_comp_chs) const
156 {
157  MatrixXd C1, C2, comp_tmp;
158 
159 // if(ctf_comp.data)
160 // delete ctf_comp.data;
161  ctf_comp.data->clear();
162 
163  if (from == to)
164  {
165  ctf_comp.data->data = MatrixXd::Identity(this->nchan, this->nchan);
166  return false;
167  }
168 
169  if (from == 0)
170  C1 = MatrixXd::Zero(this->nchan,this->nchan);
171  else
172  {
173  if (!this->make_compensator(from, C1))
174  {
175  printf("Cannot create compensator C1\n");
176  printf("Desired compensation matrix (kind = %d) not found\n",from);
177  return false;
178  }
179  }
180 
181  if (to == 0)
182  C2 = MatrixXd::Zero(this->nchan,this->nchan);
183  else
184  {
185  if (!this->make_compensator(to, C2))
186  {
187  printf("Cannot create compensator C2\n");
188  printf("Desired compensation matrix (kind = %d) not found\n",to);
189  return false;
190  }
191  }
192  //
193  // s_orig = s_from + C1*s_from = (I + C1)*s_from
194  // s_to = s_orig - C2*s_orig = (I - C2)*s_orig
195  // s_to = (I - C2)*(I + C1)*s_from = (I + C1 - C2 - C2*C1)*s_from
196  //
197  comp_tmp = MatrixXd::Identity(this->nchan,this->nchan) + C1 - C2 - C2*C1;
198 
199  qint32 k;
200  if (exclude_comp_chs)
201  {
202  VectorXi pick = MatrixXi::Zero(1,this->nchan);
203  qint32 npick = 0;
204  for (k = 0; k < this->nchan; ++k)
205  {
206  if (this->chs[k].kind != FIFFV_REF_MEG_CH)
207  {
208  pick(npick) = k;
209  ++npick;
210  }
211  }
212  if (npick == 0)
213  {
214  printf("Nothing remains after excluding the compensation channels\n");
215  return false;
216  }
217 
218  ctf_comp.data->data.resize(npick,this->nchan);
219  for (k = 0; k < npick; ++k)
220  ctf_comp.data->data.row(k) = comp_tmp.block(pick(k), 0, 1, this->nchan);
221  } else {
222  ctf_comp.data->data = comp_tmp;
223  }
224 
225  return true;
226 }

231 bool FiffInfo::make_compensator(fiff_int_t kind, MatrixXd& this_comp) const//private method
232 {
233  FiffNamedMatrix::SDPtr this_data;
234  MatrixXd presel, postsel;
235  qint32 k, col, c, ch, row, row_ch=0, channelAvailable;
236 
237  for (k = 0; k < this->comps.size(); ++k)
238  {
239  if (this->comps[k].kind == kind)
240  {
241  this_data = this->comps[k].data;
242 
243  //
244  // Create the preselector
245  //
246  presel = MatrixXd::Zero(this_data->ncol,this->nchan);
247 
248  for(col = 0; col < this_data->ncol; ++col)
249  {
250  channelAvailable = 0;
251  for (c = 0; c < this->ch_names.size(); ++c)
252  {
253  if (QString::compare(this_data->col_names.at(col),this->ch_names.at(c)) == 0)
254  {
255  ++channelAvailable;
256  ch = c;
257  }
258  }
259  if (channelAvailable == 0)
260  {
261  printf("Channel %s is not available in data\n",this_data->col_names.at(col).toUtf8().constData());
262  return false;
263  }
264  else if (channelAvailable > 1)
265  {
266  printf("Ambiguous channel %s",this_data->col_names.at(col).toUtf8().constData());
267  return false;
268  }
269  presel(col,ch) = 1.0;
270  }
271  //
272  // Create the postselector
273  //
274  postsel = MatrixXd::Zero(this->nchan,this_data->nrow);
275 
276  for (c = 0; c < this->nchan; ++c)
277  {
278  channelAvailable = 0;
279  for (row = 0; row < this_data->row_names.size(); ++row)
280  {
281  if (QString::compare(this->ch_names.at(c),this_data->row_names.at(row)) == 0)
282  {
283  ++channelAvailable;
284  row_ch = row;
285  }
286  }
287  if (channelAvailable > 1)
288  {
289  printf("Ambiguous channel %s", this->ch_names.at(c).toUtf8().constData());
290  return false;
291  }
292  else if (channelAvailable == 1)
293  {
294  postsel(c,row_ch) = 1.0;
295  }
296  }
297  this_comp = postsel*this_data->data*presel;
298  return true;
299  }
300  }
301  this_comp = defaultMatrixXd;
302  return false;
303 }
