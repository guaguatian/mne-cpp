251 void RtAveWorker::doAveraging(const MatrixXd& rawSegment)
252 {
253  //Detect trigger
254  QList<QPair<int,double> > lDetectedTriggers = DetectTrigger::detectTriggerFlanksMax(rawSegment, m_iTriggerChIndex, 0, m_fTriggerThreshold, true);
255 
256  //TODO: This does not permit the same trigger type twice in one data block
257  for(int i = 0; i < lDetectedTriggers.size(); ++i) {
258  if(!m_mapFillingBackBuffer.contains(lDetectedTriggers.at(i).second)) {
259  double dTriggerType = lDetectedTriggers.at(i).second;
260 
261  //qDebug()<<"Adding dTriggerType"<<dTriggerType;
262 
263  m_mapFillingBackBuffer[dTriggerType] = false;
264  m_mapMatDataPostIdx[dTriggerType] = 0;
265  m_mapDataPost[dTriggerType].resize(m_pFiffInfo->chs.size(), m_iPostStimSamples);
266  m_mapDataPost[dTriggerType].setZero();
267  }
268  }
269 
270  //Fill front / pre stim buffer even if no triggers have been located at all yet
271  if(m_mapFillingBackBuffer.isEmpty()) {
272  fillFrontBuffer(rawSegment, -1.0);
273  }
274 
275  //Do averaging for each trigger type
276  QMutableMapIterator<double,bool> idx(m_mapFillingBackBuffer);
277  QStringList lResponsibleTriggerTypes;
278 
279  while(idx.hasNext()) {
280  idx.next();
281 
282  double dTriggerType = idx.key();
283 
284  //Fill front / pre stim buffer
285  if(lDetectedTriggers.isEmpty()) {
286  fillFrontBuffer(rawSegment, -1.0);
287  }
288 
289  //Fill back buffer and decide when to do the data packing of the different buffers
290  if(m_mapFillingBackBuffer[dTriggerType]) {
291  if(m_mapMatDataPostIdx[dTriggerType] != m_iPostStimSamples) {
292  fillBackBuffer(rawSegment, dTriggerType);
293  }
294 
295  if(m_mapMatDataPostIdx[dTriggerType] == m_iPostStimSamples) {
296  m_mapMatDataPostIdx[dTriggerType] = 0;
297  m_mapFillingBackBuffer[dTriggerType] = false;
298  emitEvoked(dTriggerType, lResponsibleTriggerTypes);
299  }
300  } else {
301  if(lDetectedTriggers.isEmpty()) {
302  //Fill front / pre stim buffer
303  fillFrontBuffer(rawSegment, dTriggerType);
304  } else {
305  for(int i = 0; i < lDetectedTriggers.size(); ++i) {
306  if(dTriggerType == lDetectedTriggers.at(i).second) {
307  int iTriggerPos = lDetectedTriggers.at(i).first;
308 
309  //Do front buffer stuff
310  MatrixXd tempMat;
311 
312  if(iTriggerPos >= m_iPreStimSamples) {
313  tempMat = rawSegment.block(0,
314  iTriggerPos - m_iPreStimSamples,
315  rawSegment.rows(),
316  m_iPreStimSamples);
317  } else {
318  tempMat = rawSegment.block(0,
319  0,
320  rawSegment.rows(),
321  iTriggerPos);
322  }
323 
324  fillFrontBuffer(tempMat, dTriggerType);
325 
326  //Do back buffer stuff
327  if(rawSegment.cols() - iTriggerPos >= m_mapDataPost[dTriggerType].cols()) {
328  m_mapDataPost[dTriggerType] = rawSegment.block(0,
329  iTriggerPos,
330  m_mapDataPost[dTriggerType].rows(),
331  m_mapDataPost[dTriggerType].cols());
332  emitEvoked(dTriggerType, lResponsibleTriggerTypes);
333  m_mapMatDataPostIdx[dTriggerType] = 0;
334  m_mapFillingBackBuffer[dTriggerType] = false;
335  } else {
336  m_mapDataPost[dTriggerType].block(0,
337  0,
338  m_mapDataPost[dTriggerType].rows(),
339  rawSegment.cols() - iTriggerPos) = rawSegment.block(0,
340  iTriggerPos,
341  rawSegment.rows(),
342  rawSegment.cols() - iTriggerPos);
343  m_mapMatDataPostIdx[dTriggerType] = rawSegment.cols() - iTriggerPos;
344  m_mapFillingBackBuffer[dTriggerType] = true;
345  }
346 
347  //qDebug()<<"Trigger type "<<dTriggerType<<" found at "<<iTriggerPos;
348  }
349  }
350  }
351  }
352  }
353 
354  //qDebug()<<"RtAveWorker::doAveraging() - time for procesing"<<time.elapsed();
355 }

385 {

389  m_mapDataPost[dTriggerType].block(0,m_mapMatDataPostIdx[dTriggerType],m_mapDataPost[dTriggerType].rows(),iResidualCols) = data.block(0,0,data.rows(),iResidualCols);
390  } else {

400 void RtAveWorker::fillFrontBuffer(const MatrixXd &data, double dTriggerType)
401 {
402  //Init m_mapDataPre
403  if(!m_mapDataPre.contains(dTriggerType)) {
404  if(dTriggerType != -1.0) {
405  m_mapDataPre[dTriggerType] = m_mapDataPre[-1.0];
406  } else {
407  m_mapDataPre[-1.0].resize(m_pFiffInfo->chs.size(), m_iPreStimSamples);
408  m_mapDataPre[-1.0].setZero();
409  }
410  }
411 
412  if(m_mapDataPre[dTriggerType].cols() <= data.cols()) {
413  if(m_iPreStimSamples > 0 && data.cols() >= m_iPreStimSamples) {
414  m_mapDataPre[dTriggerType] = data.block(0,
415  data.cols() - m_iPreStimSamples,
416  data.rows(),
417  m_iPreStimSamples);
418  }
419  } else {
420  int residual = m_mapDataPre[dTriggerType].cols() - data.cols();
421 
422  //Copy shift data
423  m_mapDataPre[dTriggerType].block(0,
424  0,
425  m_mapDataPre[dTriggerType].rows(),
426  residual) = m_mapDataPre[dTriggerType].block(0,
427  m_mapDataPre[dTriggerType].cols() - residual,
428  m_mapDataPre[dTriggerType].rows(),
429  residual);
430 
431  //Copy new data in
432  m_mapDataPre[dTriggerType].block(0,
433  residual,
434  m_mapDataPre[dTriggerType].rows(),
435  data.cols()) = data;
436  }
437 }

442 void RtAveWorker::mergeData(double dTriggerType)
443 {
444  if(m_mapDataPre[dTriggerType].rows() != m_mapDataPost[dTriggerType].rows()) {
445  qDebug() << "RtAveWorker::mergeData - Rows of m_mapDataPre (" << m_mapDataPre[dTriggerType].rows() << ") and m_mapDataPost (" << m_mapDataPost[dTriggerType].rows() << ") are not the same. Returning.";
446  return;
447  }
448 
449  MatrixXd mergedData(m_mapDataPre[dTriggerType].rows(), m_mapDataPre[dTriggerType].cols() + m_mapDataPost[dTriggerType].cols());
450 
451  mergedData << m_mapDataPre[dTriggerType], m_mapDataPost[dTriggerType];
452 
453  //Perform artifact threshold
454  bool bArtifactDetected = false;
455 
456  if(m_bActivateThreshold && m_pFiffInfo) {
457  qDebug() << "RtAveWorker::mergeData - Doing artifact reduction for" << m_mapThresholds;
458 
459  bArtifactDetected = MNEEpochDataList::checkForArtifact(mergedData,
460  *m_pFiffInfo,
461  m_mapThresholds);
462  }
463 
464  if(!bArtifactDetected) {
465  //Add cut data to average buffer
466  m_mapStimAve[dTriggerType].append(mergedData);
467 
468  //Pop data from buffer
469  int iDiff = m_mapStimAve[dTriggerType].size() - m_iNumAverages;
470  if(iDiff > 0) {
471  for(int i = 0; i < iDiff; ++i) {
472  m_mapStimAve[dTriggerType].pop_front();
473  }
474  }
475  }
476 }

481 void RtAveWorker::generateEvoked(double dTriggerType)
482 {
483  if(m_mapStimAve[dTriggerType].isEmpty()) {
484  qDebug() << "RtAveWorker::generateEvoked - m_mapStimAve is empty for type" << dTriggerType << "Returning.";
485  return;
486  }
487 
488  //Init evoked
489  m_stimEvokedSet.info = *m_pFiffInfo.data();
490  FiffEvoked evoked;
491  evoked.setInfo(*m_pFiffInfo.data());
492  int iEvokedIdx = -1;
493 
494  for(int i = 0; i < m_stimEvokedSet.evoked.size(); ++i) {
495  if(m_stimEvokedSet.evoked.at(i).comment == QString::number(dTriggerType)) {
496  evoked = m_stimEvokedSet.evoked.at(i);
497  iEvokedIdx = i;
498  break;
499  }
500  }
501 
502  //If the evoked is not yet present add it here
503  if(iEvokedIdx == -1) {
504  evoked.baseline = m_pairBaselineSec;
505  evoked.times.resize(m_iPreStimSamples + m_iPostStimSamples);
506  evoked.times = RowVectorXf::LinSpaced(m_iPreStimSamples + m_iPostStimSamples,
507  -1*m_iPreStimSamples/m_pFiffInfo->sfreq,
508  m_iPostStimSamples/m_pFiffInfo->sfreq);
509  evoked.times[m_iPreStimSamples] = 0.0;
510  evoked.first = 0;
511  evoked.last = m_iPreStimSamples + m_iPostStimSamples;
512  evoked.comment = QString::number(dTriggerType);
513  }
514 
515  // Generate final evoked
516  MatrixXd finalAverage = MatrixXd::Zero(m_mapStimAve[dTriggerType].first().rows(), m_iPreStimSamples+m_iPostStimSamples);
517 
518  for(int i = 0; i < m_mapStimAve[dTriggerType].size(); ++i) {
519  finalAverage += m_mapStimAve[dTriggerType].at(i);
520  }
521 
522  if(!m_mapStimAve[dTriggerType].isEmpty()) {
523  finalAverage = finalAverage/m_mapStimAve[dTriggerType].size();
524  }
525 
526  if(m_bDoBaselineCorrection) {
527  finalAverage = MNEMath::rescale(finalAverage, evoked.times, m_pairBaselineSec, QString("mean"));
528  }
529 
530  evoked.data = finalAverage;
531 
532  evoked.nave = m_mapStimAve[dTriggerType].size();
533 
534  //Add new data to evoked data set
535  if(iEvokedIdx != -1) {
536  //Evoked data is already present
537  m_stimEvokedSet.evoked[iEvokedIdx] = evoked;
538  } else {
539  //Evoked data is not present yet
540  m_stimEvokedSet.evoked.append(evoked);
541  }
542 }
