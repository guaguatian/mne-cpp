230 QPair<Eigen::MatrixXd, QList<int> > Epidetect::prepareData(Eigen::MatrixXd mat)
231 {
232  MatrixXd trimmedMatrix;
233  QList<int> stimLocations;
234  QStringList badChs = m_pFiffInfo->bads;
235  QStringList chNames = m_pFiffInfo->ch_names;
236  trimmedMatrix.resize(mat.rows()- badChs.size(), mat.cols());
237 
238  int j = 0;
239 
240  for (int i=0 ; i < mat.rows(); i++)
241  {
242  QString type = m_pFiffInfo->channel_type(i);
243 
244  if (type == "stim")
245  stimLocations << i;
246  if (!(badChs.contains(chNames[i])))
247  {
248  if ((type != "stim") && (type != "ecg") && (type != "eeg"))
249  {
250  trimmedMatrix.row(j) = mat.row(i);
251  j++;
252 
253  }
254  }
255 
256  }
257 
258  qSort(stimLocations);
259 
260  trimmedMatrix.conservativeResize(j,trimmedMatrix.cols());
261  QPair<MatrixXd, QList<int> > out;
262  out.first = trimmedMatrix;
263  out.second = stimLocations;
264 
265  return out;
266

388  mu.col(0)= mu.rowwise().mean();
389  mu.col(1)= m_dvecMuFuzzyEn;

392  {
