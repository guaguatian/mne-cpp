128 void FilterData::designFilter()
129 {
130  switch(m_designMethod) {
131  case Tschebyscheff: {
132  ParksMcClellan filter(m_iFilterOrder,
133  m_dCenterFreq,
134  m_dBandwidth,
135  m_dParksWidth,
136  (ParksMcClellan::TPassType)m_Type);
137  m_dCoeffA = filter.FirCoeff;
138 
139  //fft-transform m_dCoeffA in order to be able to perform frequency-domain filtering
140  fftTransformCoeffs();
141 
142  break;
143  }
144 
145  case Cosine: {
146  CosineFilter filtercos;
147 
148  switch(m_Type) {
149  case LPF:
150  filtercos = CosineFilter(m_iFFTlength,
151  (m_dCenterFreq)*(m_sFreq/2),
152  m_dParksWidth*(m_sFreq/2),
153  (m_dCenterFreq)*(m_sFreq/2),
154  m_dParksWidth*(m_sFreq/2),
155  m_sFreq,
156  (CosineFilter::TPassType)m_Type);
157 
158  break;
159 
160  case HPF:
161  filtercos = CosineFilter(m_iFFTlength,
162  (m_dCenterFreq)*(m_sFreq/2),
163  m_dParksWidth*(m_sFreq/2),
164  (m_dCenterFreq)*(m_sFreq/2),
165  m_dParksWidth*(m_sFreq/2),
166  m_sFreq,
167  (CosineFilter::TPassType)m_Type);
168 
169  break;
170 
171  case BPF:
172  filtercos = CosineFilter(m_iFFTlength,
173  (m_dCenterFreq + m_dBandwidth/2)*(m_sFreq/2),
174  m_dParksWidth*(m_sFreq/2),
175  (m_dCenterFreq - m_dBandwidth/2)*(m_sFreq/2),
176  m_dParksWidth*(m_sFreq/2),
177  m_sFreq,
178  (CosineFilter::TPassType)m_Type);
179 
180  break;
181  }
182 
183  //This filter is designed in the frequency domain, hence the time domain impulse response need to be shortend by the users dependent number of taps
184  m_dCoeffA.resize(m_iFilterOrder);
185  m_dCoeffA.head(m_iFilterOrder/2) = filtercos.m_dCoeffA.tail(m_iFilterOrder/2);
186  m_dCoeffA.tail(m_iFilterOrder/2) = filtercos.m_dCoeffA.head(m_iFilterOrder/2);
187 
188  //Now generate the fft version of the shortened impulse response
189  fftTransformCoeffs();
190 
191  break;
192  }
193  }
194 
195  switch(m_Type) {
196  case LPF:
197  m_dLowpassFreq = 0;
198  m_dHighpassFreq = m_dCenterFreq*(m_sFreq/2);
199  break;
200 
201  case HPF:
202  m_dLowpassFreq = m_dCenterFreq*(m_sFreq/2);
203  m_dHighpassFreq = 0;
204  break;
205 
206  case BPF:
207  m_dLowpassFreq = (m_dCenterFreq + m_dBandwidth/2)*(m_sFreq/2);
208  m_dHighpassFreq = (m_dCenterFreq - m_dBandwidth/2)*(m_sFreq/2);
209  break;
210  }
211 
212 }

217 void FilterData::fftTransformCoeffs()
218 {
219  #ifdef EIGEN_FFTW_DEFAULT
220  fftw_make_planner_thread_safe();
221  #endif
222 
223  //zero-pad m_dCoeffA to m_iFFTlength
224  RowVectorXd t_coeffAzeroPad = RowVectorXd::Zero(m_iFFTlength);
225  t_coeffAzeroPad.head(m_dCoeffA.cols()) = m_dCoeffA;
226 
227  //generate fft object
228  Eigen::FFT<double> fft;
229  fft.SetFlag(fft.HalfSpectrum);
230 
231  //fft-transform filter coeffs
232  m_dFFTCoeffA = RowVectorXcd::Zero(m_iFFTlength);
233  fft.fwd(m_dFFTCoeffA,t_coeffAzeroPad);

239 RowVectorXd FilterData::applyConvFilter(const RowVectorXd& data, bool keepOverhead, CompensateEdgeEffects compensateEdgeEffects) const
240 {
241  if(data.cols()<m_dCoeffA.cols() && compensateEdgeEffects==MirrorData){
242  qDebug()<<QString("Error in FilterData: Number of filter taps(%1) bigger then data size(%2). Not enough data to perform mirroring!").arg(m_dCoeffA.cols()).arg(data.cols());
243  return data;
244  }
245 
246  //Do zero padding or mirroring depending on user input
247  RowVectorXd t_dataZeroPad = RowVectorXd::Zero(2*m_dCoeffA.cols() + data.cols());
248  RowVectorXd t_filteredTime = RowVectorXd::Zero(2*m_dCoeffA.cols() + data.cols());
249 
250  switch(compensateEdgeEffects) {
251  case MirrorData:
252  t_dataZeroPad.head(m_dCoeffA.cols()) = data.head(m_dCoeffA.cols()).reverse(); //front
253  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data; //middle
254  t_dataZeroPad.tail(m_dCoeffA.cols()) = data.tail(m_dCoeffA.cols()).reverse(); //back
255  break;
256 
257  case ZeroPad:
258  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data;
259  break;
260 
261  default:
262  t_dataZeroPad.segment(m_dCoeffA.cols(), data.cols()) = data;
263  break;
264  }
265 
266  //Do the convolution
267  for(int i=m_dCoeffA.cols(); i<t_filteredTime.cols(); i++)
268  t_filteredTime(i-m_dCoeffA.cols()) = t_dataZeroPad.segment(i-m_dCoeffA.cols(),m_dCoeffA.cols()) * m_dCoeffA.transpose();
269 
270  //Return filtered data
271  if(!keepOverhead)
272  return t_filteredTime.segment(m_dCoeffA.cols()/2, data.cols());
273 
274  return t_filteredTime.head(data.cols()+m_dCoeffA.cols());
275 }

280 RowVectorXd FilterData::applyFFTFilter(const RowVectorXd& data, bool keepOverhead, CompensateEdgeEffects compensateEdgeEffects) const
281 {
282  #ifdef EIGEN_FFTW_DEFAULT
283  fftw_make_planner_thread_safe();
284  #endif
285 
286  if(data.cols()<m_dCoeffA.cols()/2 && compensateEdgeEffects==MirrorData) {
287  qDebug()<<QString("Error in FilterData: Number of filter taps(%1) bigger then data size(%2). Not enough data to perform mirroring!").arg(m_dCoeffA.cols()).arg(data.cols());
288  return data;
289  }
290 
291  if(m_dCoeffA.cols() + data.cols()>m_iFFTlength) {
292  qDebug()<<"Error in FilterData: Number of mirroring/zeropadding size plus data size is bigger then fft length!";
293  return data;
294  }
295 
296  //Do zero padding or mirroring depending on user input
297  RowVectorXd t_dataZeroPad = RowVectorXd::Zero(m_iFFTlength);
298 
299  switch(compensateEdgeEffects) {
300  case MirrorData:
301  t_dataZeroPad.head(m_dCoeffA.cols()/2) = data.head(m_dCoeffA.cols()/2).reverse(); //front
302  t_dataZeroPad.segment(m_dCoeffA.cols()/2, data.cols()) = data; //middle
303  t_dataZeroPad.tail(m_dCoeffA.cols()/2) = data.tail(m_dCoeffA.cols()/2).reverse(); //back
304  break;
305 
306  case ZeroPad:
307  t_dataZeroPad.head(data.cols()) = data;
308  break;
309 
310  default:
311  t_dataZeroPad.head(data.cols()) = data;
312  break;
313  }
314 
315  //generate fft object
316  Eigen::FFT<double> fft;
317  fft.SetFlag(fft.HalfSpectrum);
318 
319  //fft-transform data sequence
320  RowVectorXcd t_freqData;
321  fft.fwd(t_freqData,t_dataZeroPad);
322 
323  //perform frequency-domain filtering
324  RowVectorXcd t_filteredFreq = m_dFFTCoeffA.array()*t_freqData.array();
325 
326  //inverse-FFT
327  RowVectorXd t_filteredTime;
328  fft.inv(t_filteredTime,t_filteredFreq);
329 
330  //Return filtered data
331  if(!keepOverhead)
332  return t_filteredTime.segment(m_dCoeffA.cols()/2, data.cols());
333 
334  return t_filteredTime.head(data.cols()+m_dCoeffA.cols());
335 }
