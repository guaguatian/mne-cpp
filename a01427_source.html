136  t_sDistMeasure = QString("cityblock");

143  KMeans t_kMeans(t_sDistMeasure, QString("sample"), 5);
144 
145  if(bUseWhitened)
146  {
147  t_kMeans.calculate(this->matRoiGWhitened, this->nClusters, p_RegionDataOut.roiIdx, p_RegionDataOut.ctrs, p_RegionDataOut.sumd, p_RegionDataOut.D);
148 
149  MatrixXd newCtrs = MatrixXd::Zero(p_RegionDataOut.ctrs.rows(), p_RegionDataOut.ctrs.cols());
150  for(qint32 c = 0; c < p_RegionDataOut.ctrs.rows(); ++c)
151  {
152  qint32 num = 0;
153 
154  for(qint32 idx = 0; idx < p_RegionDataOut.roiIdx.size(); ++idx)
155  {
156  if(c == p_RegionDataOut.roiIdx[idx])
157  {
158  newCtrs.row(c) += this->matRoiG.row(idx); //just take whitened to get indeces calculate centroids using the original matrix
159  ++num;
160  }
161  }
162 
163  if(num > 0)
164  newCtrs.row(c) /= num;
165  }
166  p_RegionDataOut.ctrs = newCtrs; //Replace whitened with original
167  }
168  else
169  t_kMeans.calculate(this->matRoiG, this->nClusters, p_RegionDataOut.roiIdx, p_RegionDataOut.ctrs, p_RegionDataOut.sumd, p_RegionDataOut.D);
170 
171  p_RegionDataOut.iLabelIdxOut = this->iLabelIdxIn;
172 
173  return p_RegionDataOut;
