609 double SsvepBci::MEC(MatrixXd &Y, MatrixXd &X)
610 {
611 
612  // Remove SSVEP harmonic frequencies
613  MatrixXd X_help = X.transpose()*X;
614  MatrixXd Ytilde = Y - X*X_help.inverse()*X.transpose()*Y;
615 
616  // Find eigenvalues and eigenvectors
617  SelfAdjointEigenSolver<MatrixXd> eigensolver(Ytilde.transpose()*Ytilde);
618 
619  // Determine number of channels Ns
620  int Ns;
621  VectorXd cumsum = eigensolver.eigenvalues();
622  for(int j = 1; j < eigensolver.eigenvalues().size(); j++){
623  cumsum(j) += cumsum(j - 1);
624  }
625  for(Ns = 0; Ns < eigensolver.eigenvalues().size() ; Ns++){
626  if(cumsum(Ns)/eigensolver.eigenvalues().sum() > 0.1){
627  break;
628  }
629  }
630  Ns += 1;
631 
632  // Determine spatial filter matrix W
633  MatrixXd W = eigensolver.eigenvectors().block(0, 0, eigensolver.eigenvectors().rows(), Ns);
634  for(int k = 0; k < Ns; k++){
635  W.col(k) = W.col(k)*(1/sqrt(eigensolver.eigenvalues()(k)));
636  }
637 
638  // Calcuclate channel signals
639  MatrixXd S = Y*W;
640 
641  // Calculate signal energy
642  MatrixXd P(2, Ns);
643  double power = 0;
644  for(int k = 0; k < m_iNumberOfHarmonics; k++){
645  P = X.block(0, 2*k, X.rows(), 2).transpose()*S;
646  P = P.array()*P.array();
647  power += 1 / double(m_iNumberOfHarmonics*Ns) * P.sum();
648  }
649 
650  return power;
651 }

656 double SsvepBci::CCA(MatrixXd &Y, MatrixXd &X)
657 {
658  // CCA parameter
659  int n = X.rows();
660  int p1 = X.cols();
661  int p2 = Y.cols();
662 
663  // center data sets
664  MatrixXd X_center(n, p1);
665  MatrixXd Y_center(n, p2);
666  for(int i = 0; i < p1; i++){
667  X_center.col(i) = X.col(i).array() - X.col(i).mean();
668  }
669 
670  for(int i = 0; i < p2; i++){
671  Y_center.col(i) = Y.col(i).array() - Y.col(i).mean();
672  }
673 
674  // QR decomposition
675  MatrixXd Q1, Q2;
676  ColPivHouseholderQR<MatrixXd> qr1(X_center), qr2(Y_center);
677  Q1 = qr1.householderQ() * MatrixXd::Identity(n, p1);
678  Q2 = qr2.householderQ() * MatrixXd::Identity(n, p2);
679 
680  // SVD decomposition, determine max correlation
681  JacobiSVD<MatrixXd> svd(Q1.transpose()*Q2); // ComputeThinU | ComputeThinV
682 
683  return svd.singularValues().maxCoeff();
684 }

709 void SsvepBci::ssvepBciOnSensor()
710 {
711 
712  // Wait for fiff Info if not yet received - this is needed because we have to wait until the buffers are firstly initiated in the update functions
713  while(!m_pFiffInfo_Sensor){
714  msleep(10);
715  }
716  // reset list of classifiaction results
717  MatrixXd m_matSSVEPProbabilities(m_lDesFrequencies.size(), 0);
718 
719  // Start filling buffers with data from the inputs
720  m_bProcessData = true;
721  MatrixXd t_mat = m_pBCIBuffer_Sensor->pop();
722 
723  // writing selected feature channels to the time window storage and increase the segment index
724  int writtenSamples = 0;
725  while(m_iDownSampleIndex >= m_iFormerDownSampleIndex){
726 
727  // write from t_mat to the sliding time window while doing channel select and downsampling
728  m_iFormerDownSampleIndex = m_iDownSampleIndex;
729  for(int i = 0; i < m_lElectrodeNumbers.size(); i++){
730  m_matSlidingTimeWindow(i, m_iWriteIndex) = t_mat(m_lElectrodeNumbers.at(i), m_iDownSampleIndex);
731  }
732  writtenSamples++;
733 
734  // update counter variables
735  m_iWriteIndex = (m_iWriteIndex + 1) % m_iTimeWindowLength;
736  m_iDownSampleIndex = (m_iDownSampleIndex + m_iDownSampleIncrement ) % m_iWriteSampleSize;
737 
738  }
739  m_iFormerDownSampleIndex = m_iDownSampleIndex;
740 
741  // calculate buffer between read- and write index
742  m_iReadToWriteBuffer = m_iReadToWriteBuffer + writtenSamples;
743 
744  // execute processing loop as long as there is new data to be red from the time window
745  while(m_iReadToWriteBuffer >= m_iReadSampleSize)
746  {
747  if(m_iCounter > m_iNumberOfClassBreaks)
748  {
749  // determine window size according to former counted miss classifications
750  m_iWindowSize = 10;
751  if(m_iCounter <= 50 && m_iCounter > 40){
752  m_iWindowSize = 20;
753  }
754  if(m_iCounter > 50){
755  m_iWindowSize = 40;
756  }
757 
758  // create current data matrix Y
759  MatrixXd Y;
760  readFromSlidingTimeWindow(Y);
761 
762  // create realtive timeline according to Y
763  int samples = Y.rows();
764  ArrayXd t = 2*M_PI/m_dSampleFrequency * ArrayXd::LinSpaced(samples, 1, samples);
765 
766  // Remove 50 Hz Power line signal
767  if(m_bRemovePowerLine){
768  MatrixXd Zp(samples,2);
769  ArrayXd t_PL = t*m_iPowerLine;
770  Zp.col(0) = t_PL.sin();
771  Zp.col(1) = t_PL.cos();
772  MatrixXd Zp_help = Zp.transpose()*Zp;
773  Y = Y - Zp*Zp_help.inverse()*Zp.transpose()*Y;
774  }
775 
776  qDebug() << "size of Matrix:" << Y.rows() << Y.cols();
777 
778  // apply feature extraction for all frequencies of interest
779  VectorXd ssvepProbabilities(m_lAllFrequencies.size());
780  for(int i = 0; i < m_lAllFrequencies.size(); i++)
781  {
782  // create reference signal matrix X
783  MatrixXd X(samples, 2*m_iNumberOfHarmonics);
784  for(int k = 0; k < m_iNumberOfHarmonics; k++){
785  ArrayXd t_k = t*(k+1)*m_lAllFrequencies.at(i);
786  X.col(2*k) = t_k.sin();
787  X.col(2*k+1) = t_k.cos();
788  }
789 
790  // extracting the features from the data Y with the reference signal X
791  if(m_bUseMEC){
792  ssvepProbabilities(i) = MEC(Y, X); // using Minimum Energy Combination as feature-extraction tool
793  }
794  else{
795  ssvepProbabilities(i) = CCA(Y, X); // using Canonical Correlation Analysis as feature-extraction tool
796  }
797  }
798 
799  // normalize features to probabilities and transfering it into a softmax function
800  ssvepProbabilities = m_dAlpha / ssvepProbabilities.sum() * ssvepProbabilities;
801  ssvepProbabilities = ssvepProbabilities.array().exp(); // softmax function for better distinguishability between the probabilities
802  ssvepProbabilities = 1 / ssvepProbabilities.sum() * ssvepProbabilities;
803 
804  // classify probabilites
805  int index = 0;
806  double maxProbability = ssvepProbabilities.maxCoeff(&index);
807  if(index < m_lDesFrequencies.size()){
808  //qDebug()<< "index:" << index;
809  if(m_lThresholdValues[index] < maxProbability){
810  //qDebug() << "comparison: "<< m_lThresholdValues[index] << "and" << maxProbability;
811  m_lIndexOfClassResultSensor.append(index+1);
812  }
813  else{
814  m_lIndexOfClassResultSensor.append(0);
815  }
816  }
817  else{
818  m_lIndexOfClassResultSensor.append(0);
819  }
820 
821  // clear classifiaction if it hits its threshold
822  if(m_lIndexOfClassResultSensor.size() > m_iClassListSize){
823  m_lIndexOfClassResultSensor.pop_front();
824  }
825 
826  // transfer values to matrix containing all SSVEPProabibilities of desired frequencies of one calculationstep
827  m_matSSVEPProbabilities.conservativeResize(m_lDesFrequencies.size(), m_matSSVEPProbabilities.cols() + 1);
828  m_matSSVEPProbabilities.col( m_matSSVEPProbabilities.cols() - 1) = ssvepProbabilities.head(m_lDesFrequencies.size());
829 
830  }
831 
832  // update counter and index variables
833  m_iCounter++;
834  m_iReadToWriteBuffer = m_iReadToWriteBuffer - m_iReadSampleSize;
835  m_iReadIndex = (m_iReadIndex + m_iReadSampleSize) % (m_iTimeWindowLength);
836 
837  }
838 
839  // emit classifiaction results if any classifiaction has been done
840  if(!m_lIndexOfClassResultSensor.isEmpty()){
841  // finding a classifiaction result that satisfies the number of classifiaction hits
842  for(int i = 1; (i <= m_lDesFrequencies.size()) && (!m_lIndexOfClassResultSensor.isEmpty() ); i++){
843  if(m_lIndexOfClassResultSensor.count(i) >= m_iNumberOfClassHits){
844  emit classificationResult(m_lDesFrequencies[i - 1]);
845  m_lIndexOfClassResultSensor.clear();
846  m_iCounter = 0;
847  break;
848  }
849  else{
850  emit classificationResult(0);
851  }
852  }
853  }
854 
855  // calculate and emit signal of mean probabilities
856  if(m_matSSVEPProbabilities.cols() != 0){
857  QList<double> meanSSVEPProbabilities;
858  for(int i = 0; i < m_lDesFrequencies.size(); i++){
859  meanSSVEPProbabilities << m_matSSVEPProbabilities.row(i).mean();
860  }
861  emit SSVEPprob(meanSSVEPProbabilities);
862  //qDebug() << "emit ssvep:" << meanSSVEPProbabilities;
863  }
864 
865  // change parameter and reset the time window if the change flag has been set
866  if(m_bChangeSSVEPParameterFlag){
867  changeSSVEPParameter();
868  }
869 }
