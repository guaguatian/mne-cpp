218 void Covariance::update(SCMEASLIB::Measurement::SPtr pMeasurement)
219 {
220  QSharedPointer<RealTimeMultiSampleArray> pRTMSA = pMeasurement.dynamicCast<RealTimeMultiSampleArray>();
221 
222  if(pRTMSA) {
223  //Fiff information
224  if(!m_pFiffInfo) {
225  m_pFiffInfo = pRTMSA->info();
226 
227  m_pCovarianceOutput->data()->setFiffInfo(m_pFiffInfo);
228 
229  //Set m_iEstimationSamples so that we alwyas wait for 5 secs
230  m_iEstimationSamples = m_pFiffInfo->sfreq * 5;
231  m_pRtCov = RtCov::SPtr(new RtCov(m_iEstimationSamples, m_pFiffInfo));
232  connect(m_pRtCov.data(), &RtCov::covCalculated,
233  this, &Covariance::appendCovariance);
234  }
235 
236 
237  if(m_bProcessData) {
238  MatrixXd t_mat;
239 
240  for(qint32 i = 0; i < pRTMSA->getMultiArraySize(); ++i) {
241  t_mat = pRTMSA->getMultiSampleArray()[i];
242  m_pRtCov->append(t_mat);
243  }
244  }
245  }
246 }
