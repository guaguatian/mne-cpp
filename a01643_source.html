117 VectorXd Spectrogram::gaussWindow(qint32 sample_count, qreal scale, quint32 translation)
118 {
119  VectorXd gauss = VectorXd::Zero(sample_count);
120 
121  for(qint32 n = 0; n < sample_count; n++)
122  {
