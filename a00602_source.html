115 void SensorDataTreeItem::initData(const MNEBemSurface &bemSurface,
116  const FiffInfo &fiffInfo,
117  const QString &sSensorType,
118  Qt3DCore::QEntity* p3DEntityParent)
119 {
120  if(m_bIsDataInit == true) {
121  qDebug("SensorDataTreeItem::initData - Item is already initialized");
122  }
123 
124  this->setData(0, Data3DTreeModelItemRoles::Data);
125 
126  // map passed sensor type string to fiff constant
127  fiff_int_t sensorTypeFiffConstant;
128  if (sSensorType.toStdString() == "MEG") {
129  sensorTypeFiffConstant = FIFFV_MEG_CH;
130  } else if (sSensorType.toStdString() == "EEG") {
131  sensorTypeFiffConstant = FIFFV_EEG_CH;
132  } else {
133  qDebug() << "SensorDataTreeItem::initData - unknown sensor type. Returning ...";
134  return;
135  }
136 
137  //Fill QVector with the right sensor positions, exclude bad channels
138  //Only take EEG with V as unit or MEG magnetometers with T as unit
139  QVector<Vector3f> vecSensorPos;
140  m_iUsedSensors.clear();
141  int iCounter = 0;
142  for(const FiffChInfo &info : fiffInfo.chs) {
143  if(info.kind == sensorTypeFiffConstant &&
144  (info.unit == FIFF_UNIT_T || info.unit == FIFF_UNIT_V)) {
145  vecSensorPos.push_back(info.chpos.r0);
146 
147  //save the number of the sensor
148  m_iUsedSensors.push_back(iCounter);
149  }
150  iCounter++;
151  }
152 
153  //Create bad channel idx list
154  for(const QString &bad : fiffInfo.bads) {
155  m_iSensorsBad.push_back(fiffInfo.ch_names.indexOf(bad));
156  }
157 
158  //Create InterpolationItems for CPU or GPU usage
159  if(m_bUseGPU) {
160  if(!m_pInterpolationItemGPU) {
161  m_pInterpolationItemGPU = new GpuInterpolationItem(p3DEntityParent,
162  Data3DTreeModelItemTypes::GpuInterpolationItem,
163  QStringLiteral("3D Plot"));
164 
165  m_pInterpolationItemGPU->initData(bemSurface.rr,
166  bemSurface.nn,
167  bemSurface.tris);
168 
169  QList<QStandardItem*> list;
170  list << m_pInterpolationItemGPU;
171  list << new QStandardItem(m_pInterpolationItemGPU->toolTip());
172  this->appendRow(list);
173  }
174 
175  m_pSensorRtDataWorkController->setStreamSmoothedData(false);
176 
177  connect(m_pSensorRtDataWorkController.data(), &RtSensorDataController::newInterpolationMatrixAvailable,
178  this, &SensorDataTreeItem::onNewInterpolationMatrixAvailable);
179 
180  connect(m_pSensorRtDataWorkController.data(), &RtSensorDataController::newRtRawDataAvailable,
181  this, &SensorDataTreeItem::onNewRtRawDataAvailable);
182  } else {
183  if(!m_pInterpolationItemCPU) {
184  m_pInterpolationItemCPU = new AbstractMeshTreeItem(p3DEntityParent,
185  Data3DTreeModelItemTypes::AbstractMeshItem,
186  QStringLiteral("3D Plot"));
187 
188  //Create color from curvature information with default gyri and sulcus colors
189  MatrixX4f matVertColor = AbstractMeshTreeItem::createVertColor(bemSurface.rr.rows());
190 
191  m_pInterpolationItemCPU->setMeshData(bemSurface.rr,
192  bemSurface.nn,
193  bemSurface.tris,
194  matVertColor,
195  Qt3DRender::QGeometryRenderer::Triangles);
196 
197  QList<QStandardItem*> list;
198  list << m_pInterpolationItemCPU;
199  list << new QStandardItem(m_pInterpolationItemCPU->toolTip());
200  this->appendRow(list);
201 
202  //Set material to enable sorting
203  QPointer<PerVertexPhongAlphaMaterial> pBemMaterial = new PerVertexPhongAlphaMaterial(true);
204  m_pInterpolationItemCPU->setMaterial(pBemMaterial);
205  }
206 
207  connect(m_pSensorRtDataWorkController.data(), &RtSensorDataController::newRtSmoothedDataAvailable,
208  this, &SensorDataTreeItem::onNewRtSmoothedDataAvailable);
209  }
210 
211  //Setup worker
212  m_pSensorRtDataWorkController->setInterpolationInfo(bemSurface.rr,
213  bemSurface.neighbor_vert,
214  vecSensorPos,
215  fiffInfo,
216  sensorTypeFiffConstant);
217 
218  //Init complete
219  m_bIsDataInit = true;
220 }

225 void SensorDataTreeItem::addData(const MatrixXd &tSensorData)
226 {
227  if(!m_bIsDataInit) {
228  qDebug() << "SensorDataTreeItem::addData - item has not been initialized yet!";
229  return;
230  }
231 
232  //if more data then needed is provided
233  const int iSensorSize = m_iUsedSensors.size();
234  if(tSensorData.rows() > iSensorSize)
235  {
236  MatrixXd dSmallSensorData(iSensorSize, tSensorData.cols());
237  for(int i = 0 ; i < iSensorSize; ++i)
238  {
239  //Set bad channels to zero so that the data does not corrupt while the bad channels vertex is weighted by surrounding sensors
240  if(m_iSensorsBad.contains(m_iUsedSensors[i])) {
241  dSmallSensorData.row(i).setZero();
242  } else {
243  dSmallSensorData.row(i) = tSensorData.row(m_iUsedSensors[i]);
244  }
245  }
246  //Set new data into item's data.
247  QVariant data;
248  data.setValue(dSmallSensorData);
249  this->setData(data, Data3DTreeModelItemRoles::Data);
250 
251  //Add data to worker
252  m_pSensorRtDataWorkController->addData(dSmallSensorData);
253  }
254  else
255  {
256  //Set bad channels to zero
257  MatrixXd dSmallSensorData = tSensorData;
258  for(int i = 0 ; i < dSmallSensorData.rows(); ++i)
259  {
260  if(m_iSensorsBad.contains(m_iUsedSensors[i])) {
261  dSmallSensorData.row(i).setZero();
262  }
263  }
264 
265  //Set new data into item's data.
266  QVariant data;
267  data.setValue(dSmallSensorData);
268  this->setData(data, Data3DTreeModelItemRoles::Data);
269 
270  //Add data to worker
271  m_pSensorRtDataWorkController->addData(dSmallSensorData);
272  }
273 }
