407 Eigen::VectorXi MNEMath::sort(Eigen::Matrix<T, Eigen::Dynamic, 1> &v, bool desc)
408 {
409  std::vector< std::pair<int,T> > t_vecIdxValue;
410  Eigen::VectorXi idx(v.size());
411 
412  if(v.size() > 0)
413  {
414  //fill temporal vector
415  for(qint32 i = 0; i < v.size(); ++i)
416  t_vecIdxValue.push_back(std::pair<int,T>(i, v[i]));
417 
418  //sort temporal vector
419  if(desc)
420  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairBiggerThan<T>);
421  else
422  std::sort(t_vecIdxValue.begin(), t_vecIdxValue.end(), MNEMath::compareIdxValuePairSmallerThan<T>);
423 
424  //store results
425  for(qint32 i = 0; i < v.size(); ++i)
426  {
427  idx[i] = t_vecIdxValue[i].first;
428  v[i] = t_vecIdxValue[i].second;
429  }
430  }
431 
432  return idx;
433 }

556 template<typename T>
557 void MNEMath::histcounts(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& matRawData,
558  bool bMakeSymmetrical,
559  int iClassAmount,
560  Eigen::VectorXd& vecResultClassLimits,
561  Eigen::VectorXi& vecResultFrequency,
562  double dGlobalMin,
563  double dGlobalMax)
564 {
565  if(matRawData.rows() == 0 || matRawData.cols() == 0) {
566  return;
567  }
568 
569  vecResultClassLimits.resize(iClassAmount + 1);
570  vecResultFrequency.resize(iClassAmount);
571 
572  for (int count = 0; count < iClassAmount; ++count) //initialize the vector with zero values
573  {
574  vecResultFrequency(count) = 0;
575  }
576 
577  double desiredMin,
578  desiredMax;
579  double rawMin(0.0),
580  rawMax(0.0),
581  localMin(0.0),
582  localMax(0.0);
583 
584  rawMin = matRawData.minCoeff(); //finds the raw matrix minimum value
585  rawMax = matRawData.maxCoeff(); //finds the raw matrix maximum value
586 
587  if (bMakeSymmetrical == true) //in case the user wants the histogram to have symmetrical class ranges
588  {
589  if (std::fabs(rawMin) > rawMax) //in case the negative side is larger than the positive side
590  {
591  localMax = std::fabs(rawMin); //positive side is "stretched" to the exact length as negative side
592  localMin = rawMin;
593  }
594  else if (rawMax > std::fabs(rawMin)) //in case the positive side is larger than the negative side
595  {
596  localMin = -(rawMax); //negative side is "stretched" to the exact length as positive side
597  localMax = rawMax;
598  }
599  else //in case both sides are exactly the same
600  {
601  localMin = rawMin;
602  localMax = rawMax;
603  }
604  }
605  else //in case bMakeSymmetrical == false
606  {
607  localMin = rawMin;
608  localMax = rawMax;
609  }
610  //selects either local or global range (according to user preference and input)
611  if (dGlobalMin == 0.0 && dGlobalMax == 0.0) //if global range is NOT given by the user, use local ranges
612  {
613  desiredMin = localMin;
614  desiredMax = localMax;
615  vecResultClassLimits[0] = desiredMin; //replace default value with local minimum at position 0
616  vecResultClassLimits[iClassAmount] = desiredMax; //replace default value with local maximum at position n
617  }
618  else
619  {
620  desiredMin = dGlobalMin;
621  desiredMax = dGlobalMax;
622  vecResultClassLimits(0)= desiredMin; //replace default value with global minimum at position 0
623  vecResultClassLimits(iClassAmount)= desiredMax; //replace default value with global maximum at position n
624  }
625 
626  double range = (vecResultClassLimits(iClassAmount) - vecResultClassLimits(0)), //calculates the length from maximum positive value to zero
627  dynamicUpperClassLimit;
628 
629  for (int kr = 0; kr < iClassAmount; ++kr) //dynamically initialize the upper class limit values
630  {
631  dynamicUpperClassLimit = (vecResultClassLimits(0) + (kr*(range/iClassAmount))); //generic formula to determine the upper class limit with respect to range and number of class
632  vecResultClassLimits(kr) = dynamicUpperClassLimit; //places the appropriate upper class limit value to the right position in the QVector
633  }
634 
635  for (int ir = 0; ir < matRawData.rows(); ++ir) //iterates through all columns of the data matrix
636  {
637  for (int jr = 0; jr<matRawData.cols(); ++jr) //iterates through all rows of the data matrix
638  {
639  if(matRawData(ir,jr) != 0.0) {
640  for (int kr = 0; kr < iClassAmount; ++kr) //starts iteration from 1 to iClassAmount
641  {
642  if (kr == iClassAmount-1) //used for the final iteration; if the data value is exactly the same as the final upper class limit, it will be included in the histogram
643  {
644  if (matRawData(ir,jr) >= vecResultClassLimits(kr) && matRawData(ir,jr) <= vecResultClassLimits(kr + 1)) //compares value in the matrix with lower and upper limit of each class
645  {
646  vecResultFrequency(kr) = vecResultFrequency(kr) + 1 ; //if the value fits both arguments, the appropriate class frequency is increased by 1
647  }
648  }
649  else
650  {
651  if (matRawData(ir,jr) >= vecResultClassLimits(kr) && matRawData(ir,jr) < vecResultClassLimits(kr + 1)) //compares value in the matrix with lower and upper limit of each class
652  {
653  vecResultFrequency(kr) = vecResultFrequency(kr) + 1 ; //if the value fits both arguments, the appropriate class frequency is increased by 1
654  }
655  }
656  }
657  }
658  }
659  }
660 }

665 template<typename T>
666 Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic> MNEMath::pinv(const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& a)
667 {
668  double epsilon = std::numeric_limits<double>::epsilon();
669  Eigen::JacobiSVD< Eigen::MatrixXd > svd(a ,Eigen::ComputeThinU | Eigen::ComputeThinV);
670  double tolerance = epsilon * std::max(a.cols(), a.rows()) * svd.singularValues().array().abs()(0);
671  return svd.matrixV() * (svd.singularValues().array().abs() > tolerance).select(svd.singularValues().array().inverse(),0).matrix().asDiagonal() * svd.matrixU().adjoint();
672 }
