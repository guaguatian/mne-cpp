246 void MainWindow::tabifyDockWindows()
247 {
248  //
249  // get a list of all the docks
250  //
251  QList<QDockWidget*> docks = findChildren<QDockWidget*>();
252 
253  //
254  // first, un-float all the tabs
255  //
256  std::for_each(docks.begin(), docks.end(), std::bind(&QDockWidget::setFloating, std::placeholders::_1 /* the dock widget*/, false));
257 
258  //
259  // sort them into dockWidget areas
260  //
261  QVector<QDockWidget*> topArea, leftArea, rightArea, bottomArea;
262  QVector<QVector<QDockWidget*>*> dockAreas;
263 
264  dockAreas.push_back(&topArea);
265  dockAreas.push_back(&leftArea);
266  dockAreas.push_back(&rightArea);
267  dockAreas.push_back(&bottomArea);
268 
269  std::for_each(docks.begin(), docks.end(), [&] (QDockWidget* dock)
270  {
271  if (dockWidgetArea(dock) == Qt::TopDockWidgetArea ) {topArea. push_back(dock); this->removeDockWidget(dock); dock->resize(dock->minimumSizeHint()); this->addDockWidget(Qt::TopDockWidgetArea , dock); dock->setVisible(true);}
272  else if (dockWidgetArea(dock) == Qt::LeftDockWidgetArea ) {leftArea. push_back(dock); this->removeDockWidget(dock); dock->resize(dock->minimumSizeHint()); this->addDockWidget(Qt::LeftDockWidgetArea , dock); dock->setVisible(true);}
273  else if (dockWidgetArea(dock) == Qt::RightDockWidgetArea ) {rightArea. push_back(dock); this->removeDockWidget(dock); dock->resize(dock->minimumSizeHint()); this->addDockWidget(Qt::RightDockWidgetArea , dock); dock->setVisible(true);}
274  else if (dockWidgetArea(dock) == Qt::BottomDockWidgetArea ) {bottomArea.push_back(dock); this->removeDockWidget(dock); dock->resize(dock->minimumSizeHint()); this->addDockWidget(Qt::BottomDockWidgetArea, dock); dock->setVisible(true);}
275  });
276 
277  //
278  // then, tab them all
279  //
280  for (QVector<QVector<QDockWidget*>*>::iterator areasItr = dockAreas.begin(); areasItr != dockAreas.end(); areasItr++)
281  {
282  // within each area, tab all the docks if there are more than 1
283  QVector<QDockWidget*> area = **areasItr;
284  for (int i = 1; i < area.size(); ++i) {
285  this->tabifyDockWidget(area[i-1], area[i]);
286  }
287  }
288 }
