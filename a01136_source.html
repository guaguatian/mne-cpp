442 Eigen::Matrix4d HPIFit::computeTransformation(Eigen::MatrixXd NH, Eigen::MatrixXd BT)
443 {
444  Eigen::MatrixXd xdiff, ydiff, zdiff, C, Q;
445  Eigen::Matrix4d transFinal = Eigen::Matrix4d::Identity(4,4);
446  Eigen::Matrix4d Rot = Eigen::Matrix4d::Zero(4,4);
447  Eigen::Matrix4d Trans = Eigen::Matrix4d::Identity(4,4);
448  double meanx,meany,meanz,normf;
449 
450  for(int i = 0; i < 15; ++i) {
451  // Calculate mean translation for all points -> centroid of both data sets
452  xdiff = NH.col(0) - BT.col(0);
453  ydiff = NH.col(1) - BT.col(1);
454  zdiff = NH.col(2) - BT.col(2);
455 
456  meanx = xdiff.mean();
457  meany = ydiff.mean();
458  meanz = zdiff.mean();
459 
460  // Apply translation -> bring both data sets to the same center location
461  for (int j = 0; j < BT.rows(); ++j) {
462  BT(j,0) = BT(j,0) + meanx;
463  BT(j,1) = BT(j,1) + meany;
464  BT(j,2) = BT(j,2) + meanz;
465  }
466 
467  // Estimate rotation component
468  C = BT.transpose() * NH;
469 
470  Eigen::JacobiSVD< Eigen::MatrixXd > svd(C ,Eigen::ComputeThinU | Eigen::ComputeThinV);
471 
472  Q = svd.matrixU() * svd.matrixV().transpose();
473 
474  //Handle special reflection case
475  if(Q.determinant() < 0) {
476  Q(0,2) = Q(0,2) * -1;
477  Q(1,2) = Q(1,2) * -1;
478  Q(2,2) = Q(2,2) * -1;
479  }
480 
481  // Apply rotation on translated points
482  BT = BT * Q;
483 
484  // Calculate GOF
485  normf = (NH.transpose()-BT.transpose()).norm();
486 
487  // Store rotation part to transformation matrix
488  Rot(3,3) = 1;
489  for(int j = 0; j < 3; ++j) {
490  for(int k = 0; k < 3; ++k) {
491  Rot(j,k) = Q(k,j);
492  }
493  }
494 
495  // Store translation part to transformation matrix
496  Trans(0,3) = meanx;
497  Trans(1,3) = meany;
498  Trans(2,3) = meanz;
499 
500  // Safe rotation and translation to final matrix for next iteration step
501  // This step is safe to do since we change one of the input point sets (BT)
502  // ToDo: Replace this for loop with a least square solution process
503  transFinal = Rot * Trans * transFinal;
504  }
505 
506  return transFinal;
507 }
