91 MNEEpochDataList MNEEpochDataList::readEpochs(const FiffRawData& raw,
92  const MatrixXi& events,
93  float tmin,
94  float tmax,
95  qint32 event,
96  const QMap<QString,double>& mapReject,
97  const QStringList& lExcludeChs,
98  const RowVectorXi& picks)
99 {
100  MNEEpochDataList data;
101 
102  // Select the desired events
103  qint32 count = 0;
104  qint32 p;
105  MatrixXi selected = MatrixXi::Zero(1, events.rows());
106  for (p = 0; p < events.rows(); ++p)
107  {
108  if (events(p,1) == 0 && events(p,2) == event)
109  {
110  selected(0,count) = p;
111  ++count;
112  }
113  }
114  selected.conservativeResize(1, count);
115  if (count > 0) {
116  printf("%d matching events found\n",count);
117  } else {
118  printf("No desired events found.\n");
119  return MNEEpochDataList();
120  }
121 
122  // If picks are empty, pick all
123  RowVectorXi picksNew = picks;
124  if(picks.cols() <= 0) {
125  picksNew.resize(raw.info.chs.size());
126  for(int i = 0; i < raw.info.chs.size(); ++i) {
127  picksNew(i) = i;
128  }
129  }
130 
131  fiff_int_t event_samp, from, to;
132  fiff_int_t dropCount = 0;
133  MatrixXd timesDummy;
134  MatrixXd times;
135 
136  QScopedPointer<MNEEpochData> epoch(Q_NULLPTR);
137 
138  for (p = 0; p < count; ++p) {
139  // Read a data segment
140  event_samp = events(selected(p),0);
141  from = event_samp + tmin*raw.info.sfreq;
142  to = event_samp + floor(tmax*raw.info.sfreq + 0.5);
143 
144  epoch.reset(new MNEEpochData());
145 
146  if(raw.read_raw_segment(epoch->epoch, timesDummy, from, to, picksNew)) {
147  if (p == 0) {
148  times.resize(1, to-from+1);
149  for (qint32 i = 0; i < times.cols(); ++i)
150  times(0, i) = ((float)(from-event_samp+i)) / raw.info.sfreq;
151  }
152 
153  epoch->event = event;
154  epoch->tmin = tmin;
155  epoch->tmax = tmax;
156 
157  epoch->bReject = checkForArtifact(epoch->epoch,
158  raw.info,
159  mapReject,
160  lExcludeChs);
161 
162  if (epoch->bReject) {
163  dropCount++;
164  }
165 
166  //Check if data block has the same size as the previous one
167  if(!data.isEmpty()) {
168  if(epoch->epoch.size() == data.last()->epoch.size()) {
169  data.append(MNEEpochData::SPtr(epoch.take()));//List takes ownwership of the pointer - no delete need
170  }
171  } else {
172  data.append(MNEEpochData::SPtr(epoch.take()));//List takes ownwership of the pointer - no delete need
173  }
174  } else {
175  printf("Can't read the event data segments\n");
176  }
177  }
178 
179  qDebug() << "MNEEpochDataList::readEpochs - Read a total of"<< data.size() <<"epochs of type" << event << "and marked"<< dropCount <<"for rejection";
180 
181  return data;
182 }

290  QList<int> lChTypes;

292  if(mapReject.contains("grad") ||
293  mapReject.contains("mag") ) {

295  }
296 
297  if(mapReject.contains("eeg")) {

307  }
308 
309  for(int i = 0; i < pFiffInfo.chs.size(); ++i) {
310  if(lChTypes.contains(pFiffInfo.chs.at(i).kind)
311  && !lExcludeChs.contains(pFiffInfo.chs.at(i).ch_name)
312  && !pFiffInfo.bads.contains(pFiffInfo.chs.at(i).ch_name)
313  && pFiffInfo.chs.at(i).chpos.coil_type != FIFFV_COIL_BABY_REF_MAG
314  && pFiffInfo.chs.at(i).chpos.coil_type != FIFFV_COIL_BABY_REF_MAG2) {
315  ArtifactRejectionData tempData;
316  tempData.data = data.row(i);
317 
318  switch (pFiffInfo.chs.at(i).kind) {
319  case FIFFV_MEG_CH:
320  if(pFiffInfo.chs.at(i).unit == FIFF_UNIT_T) {
321  tempData.dThreshold = mapReject["mag"];
322  } else if(pFiffInfo.chs.at(i).unit == FIFF_UNIT_T_M) {
323  tempData.dThreshold = mapReject["grad"];
324  }
325  break;
326 
327  case FIFFV_EEG_CH:
328  tempData.dThreshold = mapReject["eeg"];
329  break;
330 
331  case FIFFV_EOG_CH:
332  tempData.dThreshold = mapReject["eog"];
333  break;
334  }
335 
336  tempData.sChName = pFiffInfo.chs.at(i).ch_name;
337  lchData.append(tempData);
338  }
339  }
340 
341  if(lchData.isEmpty()) {
342  qDebug() << "MNEEpochDataList::checkForArtifact - No channels found to scan for artifacts. Do not reject. Returning.";
343 
344  return bReject;
345  }

348 
349  //Start the concurrent processing
350  QFuture<void> future = QtConcurrent::map(lchData, checkChThreshold);
351  future.waitForFinished();

353  for(int i = 0; i < lchData.size(); ++i) {
354  if(lchData.at(i).bRejected) {
355  bReject = true;
356  qDebug() << "MNEEpochDataList::checkForArtifact - Reject trial because of channel"<<lchData.at(i).sChName;
357  break;
358  }
359  }
