135 bool MNEHemisphere::add_geometry_info()
136 {
137  int k,c,p,q;
138  bool found;
139 
140  //Create neighboring triangle vector
141  neighbor_tri = QVector<QVector<int> >(this->tris.rows());
142 
143  //Create neighbor_tri information
144  for (p = 0; p < this->tris.rows(); p++) {
145  for (k = 0; k < 3; k++) {
146  this->neighbor_tri[this->tris(p,k)].append(p);
147  }
148  }
149 
150  //Create the neighboring vertices vector
151  neighbor_vert = QVector<QVector<int> >(this->np);
152 
153  for (k = 0; k < this->np; k++) {
154  for (p = 0; p < this->neighbor_tri[k].size(); p++) {
155  //Fit in the other vertices of the neighboring triangle
156  for (c = 0; c < 3; c++) {
157  int vert = this->tris(this->neighbor_tri[k][p], c);
158 
159  if (vert != k) {
160  found = false;
161 
162  for (q = 0; q < this->neighbor_vert[k].size(); q++) {
163  if (this->neighbor_vert[k][q] == vert) {
164  found = true;
165  break;
166  }
167  }
168 
169  if(!found) {
170  this->neighbor_vert[k].append(vert);
171  }
172  }
173  }
174  }
175  }
176 
177  return true;
178 }

243 bool MNEHemisphere::transform_hemisphere_to(fiff_int_t dest, const FiffCoordTrans &p_Trans)
244 {
245  FiffCoordTrans trans(p_Trans);
246 
247  if (this->coord_frame == dest)
248  {
249 // res = src;
250  return true;
251  }
252 
253  if (trans.to == this->coord_frame && trans.from == dest)
254  trans.invert_transform();
255  else if(trans.from != this->coord_frame || trans.to != dest)
256  {
257  printf("Cannot transform the source space using this coordinate transformation");//Consider throw
258  return false;
259  }
260 
261  MatrixXf t = trans.trans.block(0,0,3,4);
262 // res = src;
263  this->coord_frame = dest;
264  MatrixXf t_rr = MatrixXf::Ones(this->np, 4);
265  t_rr.block(0, 0, this->np, 3) = this->rr;
266  MatrixXf t_nn = MatrixXf::Zero(this->np, 4);
267  t_nn.block(0, 0, this->np, 3) = this->nn;
268 
269  this->rr = (t*t_rr.transpose()).transpose();
270  this->nn = (t*t_nn.transpose()).transpose();
271 
272  return true;
273 }
