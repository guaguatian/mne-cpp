222 {
223  QRect rect(xMargin+20, Margin-10, width()-2*xMargin, height() -2*Margin);
224  if(!rect.isValid()) return;
225 
226  PlotSettings settings = zoomStack[curZoom];
227  QPen quiteDark = palette().dark().color().light();
228  QPen light = palette().light().color();
229 
230  for(int i=0; i<=settings.numXTicks;++i)
231  {
232  int x = rect.left() + (i*(rect.width()-1)/settings.numXTicks);
233  double label = settings.minX + (i*settings.spanX()/settings.numXTicks);
234  painter->setPen(quiteDark);
235  painter->drawLine(x,rect.top(),x,rect.bottom());
236  painter->setPen(light);

238  painter->drawText(x,rect.bottom()+5,100,20,Qt::AlignLeft, QString::number(label));
239  }
240 
241  painter->drawText(rect.left(),rect.bottom()+20,rect.width(),20,Qt::AlignCenter, settings.xlabel);
242 
243 
244  for(int j=0; j<=settings.numYTicks;++j)
245  {
246  int y = rect.bottom() - (j*(rect.height()-1)/settings.numYTicks);
247  double label = settings.minY + (j*settings.spanY()/settings.numYTicks);
248  painter->setPen(quiteDark);
249  painter->drawLine(rect.left(),y,rect.right(),y);
250  painter->setPen(light);

278 void Plotter::drawCurve(QPainter *painter)
279 {
280  static const QColor colorForIds[6] = {Qt::red,Qt::green,Qt::blue,Qt::cyan,Qt::magenta,Qt::yellow};
281 
282  PlotSettings settings = zoomStack[curZoom];
283  QRect rect(xMargin+20, Margin-10, width()-2*xMargin, height()-2*Margin);
284  if(!rect.isValid()) return;
285 
286  painter->setClipRect(rect.adjusted(+1,+1,-1,-1));
287  QMapIterator <int, QVector<QPointF> > i(curveMap);
288  while (i.hasNext()) {
289  i.next();
290  int id = i.key();
291  QVector<QPointF> data = i.value();
292  QPolygonF polyline(data.count());
293  for(int j=0; j<data.count(); ++j){
294  double dx = data[j].x() - settings.minX;
295  double dy = data[j].y() - settings.minY;
296  double x = rect.left() + (dx*(rect.width()-1)/settings.spanX());
297  double y = rect.bottom() - (dy *(rect.height()-1)/settings.spanY());
298  polyline[j] = QPointF(x,y);
299  }
300  painter->setPen(colorForIds[uint(id) % 6]);
301  painter->drawPolyline(polyline);
302  }
303 }

322 void PlotSettings::adjustAxis(double &min, double &max, int &numTicks)
323 {
324  const int MinTicks = 4;
325  double grossStep = (max-min)/ MinTicks;
326  double step = std::pow(10.0, std::floor(std::log10(grossStep)));
327 
328  if(5*step<grossStep){
329  step *= 5;
330  }
331  else if(2*step < grossStep){
332  step *= 2;
333  }
334 
335  numTicks = int(std::ceil(max/step) - std::floor(min/step));
336  if(numTicks < MinTicks)
337  numTicks = MinTicks;
338  min = std::floor(min/step)*step;
339  max = std::ceil(max/step)*step;
340 }
