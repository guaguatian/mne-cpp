242 template<typename _Tp>
243 inline void CircularMatrixBuffer<_Tp>::push(const Matrix<_Tp, Dynamic, Dynamic>* pMatrix)
244 {
245  if(!m_bPause)
246  {
247  unsigned int t_size = pMatrix->size();
248  if(t_size == m_uiRows*m_uiCols)
249  {
250  m_pFreeElements->acquire(t_size);
251  for(unsigned int i = 0; i < t_size; ++i)
252  m_pBuffer[mapIndex(m_iCurrentWriteIndex)] = pMatrix->data()[i];
253  m_pUsedElements->release(t_size);
254  }
255 
256  else {
257  printf("Error: Matrix not appended to CircularMatrixBuffer - wrong dimensions\n");
258  }
