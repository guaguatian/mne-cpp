175 void SsvepBciConfigurationWidget::setSSVEPProbabilities(MyQList SSVEP){
176 
177  // determine scale for threshold status bar
178  if(m_bInitThresholdDisplay){
179  // reset scale for threshold values & status bar
180  m_dMaxProbValue = *std::max_element(SSVEP.begin(), SSVEP.end());
181  m_dMinProbValue = *std::min_element(SSVEP.begin(), SSVEP.end());
182  resetThresholdValues();
183  m_bInitThresholdDisplay = false;
184  }
185  else{
186  // update scale for threshold status bar
187  double min = *std::min_element(SSVEP.begin(), SSVEP.end());
188  double max = *std::max_element(SSVEP.begin(), SSVEP.end());
189  m_dMinProbValue = min < m_dMinProbValue ? min : m_dMinProbValue;
190  m_dMaxProbValue = max > m_dMaxProbValue ? max : m_dMaxProbValue;
191  }
192 
193  // filling rest of SSVEP with zeros if size of SSVEP < 5
194  for(int i = SSVEP.size(); i < 5; i++){
195  SSVEP << m_dMinProbValue;
196  }
197 
198  // scale SSVEP values for status bar
199  QList<int> values;
200  for(int i = 0; i < SSVEP.size(); i++){
201  values << int((SSVEP.at(i) - m_dMinProbValue) / ( m_dMaxProbValue - m_dMinProbValue ) * 100);
202  }
203 
204  // assign SSVEP values to the status bar
205  ui->m_ProgressBar_Threshold1->setValue(values[0]);
206  ui->m_ProgressBar_Threshold2->setValue(values[1]);
207  ui->m_ProgressBar_Threshold3->setValue(values[2]);
208  ui->m_ProgressBar_Threshold4->setValue(values[3]);
209  ui->m_ProgressBar_Threshold5->setValue(values[4]);
210 
211  // assign SSVEP values to the labels
212  ui->m_Label_SSVEP1->setText(QString::number(SSVEP[0]));
213  ui->m_Label_SSVEP2->setText(QString::number(SSVEP[1]));
214  ui->m_Label_SSVEP3->setText(QString::number(SSVEP[2]));
215  ui->m_Label_SSVEP4->setText(QString::number(SSVEP[3]));

236 void SsvepBciConfigurationWidget::initSelectedChannelsSensor()
237 {
238  // Read electrode pinnig scheme from file and initialise List and store in QMap in BCI object
239  QString path;
240  path.prepend(m_pSsvepBci->getSsvepBciResourcePath());
241  path.append("Pinning_Scheme_Duke_Dry_64.txt"); //Brain_Amp_presentation.txt
242  QFile file(path);
243  if(!file.open(QIODevice::ReadOnly | QIODevice::Text)){
244  return;
245  }
246 
247  //Start reading from file
248  m_vAvailableChannelsSensor.clear();
249  QMap<QString, int> mapElectrodePinningScheme;
250 
251  QTextStream in(&file);
252  while(!in.atEnd())
253  {
254  QString line = in.readLine();
255  QStringList list_temp = line.split(QRegExp("\\s+"));
256 
257  if(list_temp.size() >= 2)
258  mapElectrodePinningScheme.insert(list_temp.at(1), list_temp.at(0).toInt()-1); // Decrement by 1 because channels in matrix start with 0
259  m_vAvailableChannelsSensor.append(list_temp.at(1));
260  }
261  file.close();
262  m_pSsvepBci->m_mapElectrodePinningScheme = mapElectrodePinningScheme;
263 
264  // Remove default items from list
265  for(int i=0; i<m_pSsvepBci->m_slChosenChannelsSensor.size(); i++){
266  m_vAvailableChannelsSensor.removeAt(m_vAvailableChannelsSensor.indexOf(m_pSsvepBci->m_slChosenChannelsSensor.at(i)));
267  }
268 
269  ui->m_listWidget_AvailableChannelsOnSensorLevel->addItems(m_vAvailableChannelsSensor);
270  ui->m_listWidget_ChosenChannelsOnSensorLevel->addItems(m_pSsvepBci->m_slChosenChannelsSensor);
271 }
