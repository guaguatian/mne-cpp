196 void MNE::calcFiffInfo()
197 {
198  QMutexLocker locker(&m_qMutex);
199 
200  if(m_qListCovChNames.size() > 0 && m_pFiffInfoInput && m_pFiffInfoForward)
201  {
202  qDebug() << "MNE::calcFiffInfoFiff - Infos available";
203 
204 // qDebug() << "MNE::calcFiffInfo - m_qListCovChNames" << m_qListCovChNames;
205 // qDebug() << "MNE::calcFiffInfo - m_pFiffInfoForward->ch_names" << m_pFiffInfoForward->ch_names;
206 // qDebug() << "MNE::calcFiffInfo - m_pFiffInfoInput->ch_names" << m_pFiffInfoInput->ch_names;
207 
208  // Align channel names of the forward solution to the incoming averaged (currently acquired) data
209  // Find out whether the forward solution depends on only MEG, EEG or both MEG and EEG channels
210  QStringList forwardChannelsTypes;
211  m_pFiffInfoForward->ch_names.clear();
212  int counter = 0;
213 
214  for(qint32 x = 0; x < m_pFiffInfoForward->chs.size(); ++x) {
215  if(forwardChannelsTypes.contains("MEG") && forwardChannelsTypes.contains("EEG"))
216  break;
217 
218  if(m_pFiffInfoForward->chs[x].kind == FIFFV_MEG_CH && !forwardChannelsTypes.contains("MEG"))
219  forwardChannelsTypes<<"MEG";
220 
221  if(m_pFiffInfoForward->chs[x].kind == FIFFV_EEG_CH && !forwardChannelsTypes.contains("EEG"))
222  forwardChannelsTypes<<"EEG";
223  }
224 
225  //If only MEG channels are used
226  if(forwardChannelsTypes.contains("MEG") && !forwardChannelsTypes.contains("EEG")) {
227  for(qint32 x = 0; x < m_pFiffInfoInput->chs.size(); ++x)
228  {
229  if(m_pFiffInfoInput->chs[x].kind == FIFFV_MEG_CH) {
230  m_pFiffInfoForward->chs[counter].ch_name = m_pFiffInfoInput->chs[x].ch_name;
231  m_pFiffInfoForward->ch_names << m_pFiffInfoInput->chs[x].ch_name;
232  counter++;
233  }
234  }
235  }
236 
237  //If only EEG channels are used
238  if(!forwardChannelsTypes.contains("MEG") && forwardChannelsTypes.contains("EEG")) {
239  for(qint32 x = 0; x < m_pFiffInfoInput->chs.size(); ++x)
240  {
241  if(m_pFiffInfoInput->chs[x].kind == FIFFV_EEG_CH) {
242  m_pFiffInfoForward->chs[counter].ch_name = m_pFiffInfoInput->chs[x].ch_name;
243  m_pFiffInfoForward->ch_names << m_pFiffInfoInput->chs[x].ch_name;
244  counter++;
245  }
246  }
247  }
248 
249  //If both MEG and EEG channels are used
250  if(forwardChannelsTypes.contains("MEG") && forwardChannelsTypes.contains("EEG")) {
251  //qDebug()<<"MNE::calcFiffInfo - MEG EEG fwd solution";
252  for(qint32 x = 0; x < m_pFiffInfoInput->chs.size(); ++x)
253  {
254  if(m_pFiffInfoInput->chs[x].kind == FIFFV_MEG_CH || m_pFiffInfoInput->chs[x].kind == FIFFV_EEG_CH) {
255  m_pFiffInfoForward->chs[counter].ch_name = m_pFiffInfoInput->chs[x].ch_name;
256  m_pFiffInfoForward->ch_names << m_pFiffInfoInput->chs[x].ch_name;
257  counter++;
258  }
259  }
260  }
261 
262  //Pick only channels which are present in all data structures (covariance, evoked and forward)
263  QStringList tmp_pick_ch_names;
264  foreach (const QString &ch, m_pFiffInfoForward->ch_names)
265  {
266  if(m_pFiffInfoInput->ch_names.contains(ch))
267  tmp_pick_ch_names << ch;
268  }
269  m_qListPickChannels.clear();
270 
271  foreach (const QString &ch, tmp_pick_ch_names)
272  {
273  if(m_qListCovChNames.contains(ch))
274  m_qListPickChannels << ch;
275  }
276  RowVectorXi sel = m_pFiffInfoInput->pick_channels(m_qListPickChannels);
277 
278  //qDebug() << "MNE::calcFiffInfo - m_qListPickChannels.size()" << m_qListPickChannels.size();
279  //qDebug() << "MNE::calcFiffInfo - m_qListPickChannels" << m_qListPickChannels;
280 
281  m_pFiffInfo = QSharedPointer<FiffInfo>(new FiffInfo(m_pFiffInfoInput->pick_info(sel)));
282 
283  m_pRTSEOutput->data()->setFiffInfo(m_pFiffInfo);
284 
285  qDebug() << "MNE::calcFiffInfo - m_pFiffInfo" << m_pFiffInfo->ch_names;
286  }
287 }

397 void MNE::updateRTMSA(SCMEASLIB::Measurement::SPtr pMeasurement)
398 {
399  QSharedPointer<RealTimeMultiSampleArray> pRTMSA = pMeasurement.dynamicCast<RealTimeMultiSampleArray>();
400 
401  if(pRTMSA) {
402  if(!m_bReceiveData) {
403  return;
404  }
405 
406  //Check if buffer initialized
407  if(!m_pMatrixDataBuffer) {
408  m_pMatrixDataBuffer = CircularMatrixBuffer<double>::SPtr(new CircularMatrixBuffer<double>(64, pRTMSA->getNumChannels(), pRTMSA->getMultiSampleArray()[0].cols()));
409  }
410 
411  //Fiff Information of the RTMSA
412  if(!m_pFiffInfoInput) {
413  m_pFiffInfoInput = pRTMSA->info();
414  m_iNumAverages = 1;
415  }
416 
417  if(m_bProcessData) {
418  for(qint32 i = 0; i < pRTMSA->getMultiSampleArray().size(); ++i) {
419  // Check for artifacts
420  QMap<QString,double> mapReject;
421  mapReject.insert("eog", 150e-06);
422 
423  bool bArtifactDetected = MNEEpochDataList::checkForArtifact(pRTMSA->getMultiSampleArray()[i],
424  *m_pFiffInfoInput,
425  mapReject);
426 
427  if(!bArtifactDetected) {
428  m_pMatrixDataBuffer->push(&pRTMSA->getMultiSampleArray()[i]);
429  } else {
430  qDebug() << "MNE::updateRTMSA - Reject data block";
431  }
432 
433  }
434  }
435  }
436 }

468 void MNE::updateRTE(SCMEASLIB::Measurement::SPtr pMeasurement)
469 {
470  QSharedPointer<RealTimeEvokedSet> pRTES = pMeasurement.dynamicCast<RealTimeEvokedSet>();
471 
472  if(!pRTES) {
473  return;
474  }
475 
476  QMutexLocker locker(&m_qMutex);
477 
478  QStringList lResponsibleTriggerTypes = pRTES->getResponsibleTriggerTypes();
479 
480  if(m_pMinimumNormSettingsView) {
481  m_pMinimumNormSettingsView->setTriggerTypes(lResponsibleTriggerTypes);
482  }
483 
484  if(!m_bReceiveData || !lResponsibleTriggerTypes.contains(m_sAvrType)) {
485  return;
486  }
487 
488  FiffEvokedSet::SPtr pFiffEvokedSet = pRTES->getValue();
489 
490  //Fiff Information of the evoked
491  if(!m_pFiffInfoInput && pFiffEvokedSet->evoked.size() > 0) {
492  for(int i = 0; i < pFiffEvokedSet->evoked.size(); ++i) {
493  if(pFiffEvokedSet->evoked.at(i).comment == m_sAvrType) {
494  m_pFiffInfoInput = QSharedPointer<FiffInfo>(new FiffInfo(pFiffEvokedSet->evoked.at(i).info));
495  break;
496  }
497  }
498  }
499 
500  if(m_bProcessData) {
501  for(int i = 0; i < pFiffEvokedSet->evoked.size(); ++i) {
502  if(pFiffEvokedSet->evoked.at(i).comment == m_sAvrType) {
503  //qDebug()<<"MNE::updateRTE - average found type" << m_sAvrType;
504  m_qVecFiffEvoked.push_back(pFiffEvokedSet->evoked.at(i).pick_channels(m_qListPickChannels));
505  break;
506  }
507  }
508  }

581  m_bReceiveData = true;

598

619 // 0.05,

621 // true);

624 // m_invOp = MNEInverseOperator(*m_pFiffInfoInput,
625 // *m_pClusteredFwd,
626 // noise_cov,
627 // 0.2f,

629

631 // double lambda2 = 1.0 / pow(snr, 2); //ToDo estimate lambda using covariance
632 // QString method("dSPM"); //"MNE" | "dSPM" | "sLORETA"
633 // m_pMinimumNorm = MinimumNorm::SPtr(new MinimumNorm(m_invOp,
634 // lambda2,

671  }
672

674  tstep = 1.0f / m_pFiffInfoInput->sfreq;
675
