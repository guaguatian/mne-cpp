109 void ButterflySceneItem::paintAveragePaths(QPainter *painter)
110 {
111  //Create path for all channels
112  for(int i = 0; i < m_pFiffInfo->chs.size() ;i++) {
113 
114  FiffChInfo fiffChInfoTemp = m_pFiffInfo->chs.at(i);
115 
116  if(m_pFiffInfo->bads.contains(fiffChInfoTemp.ch_name) == false) {
117  //Only plot EEG or MEG channels
118  if((fiffChInfoTemp.kind == FIFFV_MEG_CH && m_iSetKind == FIFFV_MEG_CH && fiffChInfoTemp.unit == FIFF_UNIT_T_M && m_iSetUnit == FIFF_UNIT_T_M) || //MEG grad
119  (fiffChInfoTemp.kind == FIFFV_MEG_CH && m_iSetKind == FIFFV_MEG_CH && fiffChInfoTemp.unit == FIFF_UNIT_T && m_iSetUnit == FIFF_UNIT_T) || //MEG mag
120  (fiffChInfoTemp.kind == FIFFV_EEG_CH && m_iSetKind == FIFFV_EEG_CH)) { //EEG
121  //Determine channel scaling
122  double dMaxValue = 1e-09;
123  switch(fiffChInfoTemp.kind) {
124  case FIFFV_MEG_CH: {
125  if(fiffChInfoTemp.unit == FIFF_UNIT_T_M) {
126  dMaxValue = m_scaleMap["MEG_grad"];
127  }
128  else if(fiffChInfoTemp.unit == FIFF_UNIT_T)
129  dMaxValue = m_scaleMap["MEG_mag"];
130  break;
131  }
132  case FIFFV_EEG_CH: {
133  dMaxValue = m_scaleMap["MEG_EEG"];
134  break;
135  }
136  }
137 
138  //Get data pointer for the current channel
139  const double* averageData = m_lAverageData.first;
140  int totalCols = m_lAverageData.second; //equals to the number of samples stored in the data matrix
141 
142  //Calculate downsampling factor of averaged data in respect to the items width
143  int dsFactor;
144  QRectF boundingRect = this->boundingRect();
145  totalCols / boundingRect.width()<1 ? dsFactor = 1 : dsFactor = totalCols / boundingRect.width();
146 
147  //Calculate scaling value
148  double dScaleY = (boundingRect.height())/(2*dMaxValue);
149 
150  //Setup the painter
151  QPainterPath path = QPainterPath(QPointF(boundingRect.x(), *(averageData+(0*m_pFiffInfo->chs.size())+i) * -dScaleY));
152  QPen pen;
153  pen.setStyle(Qt::SolidLine);
154  if(!m_cAverageColors.isEmpty() && i<m_cAverageColors.size())
155  pen.setColor(m_cAverageColors.at(i));
156  pen.setWidthF(0.25);
157  painter->setPen(pen);
158 
159  //Generate plot path
160  QPointF qSamplePosition;
161 
162  for(int u = 0; u < totalCols && path.elementCount() <= boundingRect.width(); u += dsFactor) {
163  //evoked matrix is stored in column major
164  double val = (*(averageData+(u*m_pFiffInfo->chs.size())+i) * dScaleY);
165 
166  qSamplePosition.setY(-val);
167  qSamplePosition.setX(path.currentPosition().x()+1);
168 
169  path.lineTo(qSamplePosition);
170  }
171

174  }
175  }
