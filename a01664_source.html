104 {
105  MatrixXi t_Y = MatrixXi::Zero(sample_size, 1);

107  t_Y(i,0) = rand() % num_classes;
108  }

111  std::normal_distribution<float> distribution(0.0,1.0);
112

115  for(int j = 0; j < X.cols(); ++j) {
