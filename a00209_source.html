313 void Spline::splitCoefficientAndExponent (const Eigen::Matrix<T, Eigen::Dynamic, Eigen::Dynamic>& matClassLimitData,
314  int iClassAmount,
315  Eigen::VectorXd& vecCoefficientResults,
316  Eigen::VectorXi& vecExponentValues)
317 {
318  vecCoefficientResults.resize(iClassAmount + 1);
319  vecExponentValues.resize(iClassAmount + 1);
320  double originalValue(0.0),
321  limitDisplayValue(0.0),
322  doubleExponentValue(0.0);
323  int limitExponentValue(0);
324  for (int ir = 0; ir <= iClassAmount; ++ir)
325  {
326  originalValue = matClassLimitData(ir);
327  if (originalValue == 0.0) //mechanism to guard against evaluation of log(0.0) which is infinity
328  {
329  doubleExponentValue = 0.0;
330  }
331 
332  else
333  {
334  doubleExponentValue = log10(std::fabs(originalValue)); //return the exponent value in double
335  }
336 
337  limitExponentValue = round(doubleExponentValue); //round the exponent value to the nearest signed integer
338  limitDisplayValue = originalValue * (pow(10,-(limitExponentValue))); //display value is derived from multiplying class limit with inverse 10 to the power of negative exponent
339  vecCoefficientResults(ir) = limitDisplayValue; //append the display value to the return vector
340  vecExponentValues(ir) = limitExponentValue; //append the exponent value to the return vector
341  }
342 
343  int lowestExponentValue{0},
344  highestExponentValue{0};
345 
346  for (int ir = 0; ir <= iClassAmount; ++ir)
347  {
348  if (vecExponentValues(ir) < lowestExponentValue)
349  {
350  lowestExponentValue = vecExponentValues(ir); //find lowest exponent value to normalize display values for negative exponent
351  }
352  if (vecExponentValues(ir) > highestExponentValue) //find highest exponent value to normalize display values for positive exponent
353  {
354  highestExponentValue = vecExponentValues(ir);
355  }
356  }
357 
358  if (highestExponentValue > 0)
359  {
360  for (int ir = 0; ir <= iClassAmount; ++ir)
361  {
362  while (vecExponentValues(ir) < highestExponentValue) //normalize the values by multiplying the display value by 10 and reducing the exponentValue by 1 until exponentValue reach the lowestExponentValue
363  {
364  vecCoefficientResults(ir) = vecCoefficientResults(ir) / 10;
365  vecExponentValues(ir)++;
366  }
367  }
368  }
369 
370  if (lowestExponentValue < 0)
371  {
372  for (int ir = 0; ir <= iClassAmount; ++ir)
373  {
374  while (vecExponentValues(ir) > lowestExponentValue)
375  {
376  vecCoefficientResults(ir) = vecCoefficientResults(ir) * 10;
377  vecExponentValues(ir)--;
378  }
379  }
380  }
381 }
