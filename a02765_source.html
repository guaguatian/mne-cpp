252 bool Natus::start()
253 {
254 // //Check if the thread is already or still running.
255 // //This can happen if the start button is pressed immediately after the stop button was pressed.
256 // //In this case the stopping process is not finished yet but the start process is initiated.
257 // if(this->isRunning()) {
258 // this->wait();
259 // }
260 
261  m_bIsRunning = true;
262 
263  //Setup fiff info before setting up the RMTSA because we need it to init the RTMSA
264  setUpFiffInfo();
265 
266  //Set the channel size of the RMTSA - this needs to be done here and NOT in the init() function because the user can change the number of channels during runtime
267  m_pRMTSA_Natus->data()->initFromFiffInfo(m_pFiffInfo);
268  m_pRMTSA_Natus->data()->setMultiArraySize(1);
269 
270  QThread::start();
271 
272  // Start the producer
273  m_pNatusProducer = QSharedPointer<NatusProducer>::create(m_iSamplesPerBlock, m_iNumberChannels);
274  m_pNatusProducer->moveToThread(&m_pProducerThread);
275  connect(m_pNatusProducer.data(), &NatusProducer::newDataAvailable,
276  this, &Natus::onNewDataAvailable,
277  Qt::DirectConnection);
278  m_pProducerThread.start();
279 
280  return true;
281 }
