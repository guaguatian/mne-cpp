124 bool MNEBemSurface::addTriangleData()
125 {
126  //
127  // Main triangulation
128  //
129  printf("\tCompleting triangulation info...");
130  this->tri_cent = MatrixX3d::Zero(this->ntri,3);
131  this->tri_nn = MatrixX3d::Zero(this->ntri,3);
132  this->tri_area = VectorXd::Zero(this->ntri);
133 
134  Matrix3d r;
135  Vector3d a, b;
136  int k = 0;
137  float size = 0;
138  for (qint32 i = 0; i < this->ntri; ++i)
139  {
140  for ( qint32 j = 0; j < 3; ++j)
141  {
142  k = this->tris(i, j);
143 
144  r(j,0) = this->rr(k, 0);
145  r(j,1) = this->rr(k, 1);
146  r(j,2) = this->rr(k, 2);
147 
148  this->tri_cent(i, 0) += this->rr(k, 0);
149  this->tri_cent(i, 1) += this->rr(k, 1);
150  this->tri_cent(i, 2) += this->rr(k, 2);
151  }
152  this->tri_cent.row(i) /= 3.0f;
153 
154  //cross product {cross((r2-r1),(r3-r1))}
155  a = r.row(1) - r.row(0 );
156  b = r.row(2) - r.row(0);
157  this->tri_nn(i,0) = a(1)*b(2)-a(2)*b(1);
158  this->tri_nn(i,1) = a(2)*b(0)-a(0)*b(2);
159  this->tri_nn(i,2) = a(0)*b(1)-a(1)*b(0);
160 
161  //area
162  size = this->tri_nn.row(i)*this->tri_nn.row(i).transpose();
163  size = std::pow(size, 0.5f );
164 
165  this->tri_area(i) = size/2.0f;
166  this->tri_nn.row(i) /= size;
167  }
168 
169  std::fstream doc("./Output/tri_area.dat", std::ofstream::out | std::ofstream::trunc);
170  if(doc) // if succesfully opened
171  {
172  // instructions
173  doc << this->tri_area << "\n";
174  doc.close();
175  }
176 
177  printf("Adding additional geometry info\n");
178  add_geometry_info();

180  printf("[done]\n");
