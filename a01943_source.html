240 bool EventWindow::event(QEvent * event)
241 {
242  //On resize event center marker again
243  if(event->type() == QEvent::Resize) {
244  qDebug()<<"resize";
245  jumpToEvent(ui->m_tableView_eventTableView->selectionModel()->currentIndex(), QModelIndex());
246  }
247 
248  //Delete selected row on delete key press event
249  if(event->type() == QEvent::KeyPress && ui->m_tableView_eventTableView->hasFocus()) {
250  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
251  if(keyEvent->key() == Qt::Key_Delete)
252  removeEventfromEventModel();
253  }
254 
255  return QDockWidget::event(event);

261 void EventWindow::jumpToEvent(const QModelIndex & current, const QModelIndex & previous)
262 {
263  Q_UNUSED(previous);
264 
265  if(ui->m_checkBox_activateEvents->isChecked()) {
266  //Always get the first column 0 (sample) of the model - Note: Need to map index from sorting model back to source model
267  QModelIndex index = m_pEventModel->index(current.row(), 0);
268 
269  //Get the sample value
270  int sample = m_pEventModel->data(index, Qt::DisplayRole).toInt();
271 
272  //Jump to sample - put sample in the middle of the view - the viewport holds the width of the are which is changed through scrolling
273  int rawTableViewColumnWidth = m_pMainWindow->m_pDataWindow->getDataTableView()->viewport()->width();
274 
275  if(sample-rawTableViewColumnWidth/2 < rawTableViewColumnWidth/2) //events lie in the first half of the data window at the beginning of the loaded data -> cannot centralize view on event
276  m_pMainWindow->m_pDataWindow->getDataTableView()->horizontalScrollBar()->setValue(0);
277  else if(sample+rawTableViewColumnWidth/2 > m_pMainWindow->m_pDataWindow->getDataModel()->lastSample()-rawTableViewColumnWidth/2) //events lie in the last half of the data window at the end of the loaded data -> cannot centralize view on event
278  m_pMainWindow->m_pDataWindow->getDataTableView()->horizontalScrollBar()->setValue(m_pMainWindow->m_pDataWindow->getDataTableView()->maximumWidth());
279  else //centralize view on event
280  m_pMainWindow->m_pDataWindow->getDataTableView()->horizontalScrollBar()->setValue(sample-rawTableViewColumnWidth/2);
281 
282  qDebug()<<"Jumping to Event at sample "<<sample<<"rawTableViewColumnWidth"<<rawTableViewColumnWidth;
283 
284  m_pMainWindow->m_pDataWindow->updateDataTableViews();
285  }
286 }
