100 void LSLAdapterProducer::readStream()
101 {
102  // check if we have a stream info
103  if (m_bHasStreamInfo == false) {
104  qDebug() << "[LSLAdapterProducer::readStream] No stream info was supplied !";
105  return;
106  }
107 
108  // start to stream: build a stream inlet
109  try {
110  m_StreamInlet = new lsl::stream_inlet(m_StreamInfo);
111  m_StreamInlet->open_stream();
112  }
113  catch (std::exception& e) {
114  qDebug() << "[LSLAdapterProducer::readStream] Something went wrong when trying to open LSL stream inlet: " << e.what();
115  }
116 
117  m_bIsRunning = true;
118  while(m_bIsRunning) {
119  try {
120  if(m_StreamInlet->samples_available() == false) {
121  // save CPU time, then check again
122  QThread::msleep(5);
123  continue;
124  }
125  std::vector<std::vector<float>> chunk = m_StreamInlet->pull_chunk<float>();
126 
127  // append chunk to buffer
128  m_vBufferedSamples.insert(std::end(m_vBufferedSamples), std::begin(chunk), std::end(chunk));
129 
130  // check if we can output another block
131  if(m_vBufferedSamples.size() >= m_iOutputBlockSize) {
132  Eigen::MatrixXd matOutput(m_StreamInfo.channel_count(), m_iOutputBlockSize);
133 
134  // copy samples
135  for(int iSampleIdx = 0; iSampleIdx < m_iOutputBlockSize; ++iSampleIdx) {
136  for(int iChannelIdx = 0; iChannelIdx < m_StreamInfo.channel_count(); ++iChannelIdx) {
137  matOutput(iChannelIdx, iSampleIdx) = static_cast<double>(m_vBufferedSamples[iSampleIdx][iChannelIdx]);
138  }
139  }
140 
141  // remove copied samples
142  for(int i = 0; i < m_iOutputBlockSize; ++i) {
143  m_vBufferedSamples.erase(m_vBufferedSamples.begin());
144  }
145 
146  // publish new block
147  m_pRTMSA->data()->setValue(matOutput);
148  }
149  }
150  catch (std::exception& e) {
151  qDebug() << "[LSLAdapterProducer::readStream] Something went wrong while streaming data: " << e.what();
152  m_bIsRunning = false;
153  }
154  }
155 
156  // cleanup: close stream
157  try {
158  m_StreamInlet->close_stream();
159  }
160  catch (std::exception& e) {
161  qDebug() << "[LSLAdapterProducer::readStream] Something went wrong when trying to close LSL stream: " << e.what();
162  }
163 
164  emit finished();
165 }
