101 Eigen::VectorXd sineWaveGenerator(double amplitude, double xStep, int xNow, int xEnd)
102 {
103  unsigned int iterateAmount = ceil((xEnd-xNow)/xStep);
104  Eigen::VectorXd sineWaveResultOriginal;
105  sineWaveResultOriginal.resize(iterateAmount+1);
106  Eigen::VectorXd sineWaveResult = sineWaveResultOriginal.transpose();
107  double sineResult;
108  double omega = 2.0*M_PI;
109  int iterateCount = 0;
110  for (double step = xNow; step < xEnd; step +=xStep)
111  {
112  sineResult = amplitude* (sin(omega * step));
113  sineWaveResult(iterateCount) = sineResult;
114  iterateCount++;
115  }
116  return sineWaveResult;
117 }
