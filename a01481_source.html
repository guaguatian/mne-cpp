95 MatrixXd RtFilter::filterDataBlock(const MatrixXd& matDataIn,
96  int iOrder,
97  const RowVectorXi &vecPicks,
98  const QList<FilterData>& lFilterData)
99 {
100  //Initialise the overlay matrix
101  if(m_matOverlap.cols() != iOrder || m_matOverlap.rows() < matDataIn.rows()) {
102  m_matOverlap.resize(matDataIn.rows(), iOrder);
103  m_matOverlap.setZero();
104  }
105 
106  if(m_matDelay.cols() != iOrder/2 || m_matOverlap.rows() < matDataIn.rows()) {
107  m_matDelay.resize(matDataIn.rows(), iOrder/2);
108  m_matDelay.setZero();
109  }
110 
111  //Resize output matrix to match input matrix
112  MatrixXd matDataOut = matDataIn;
113 
114  //Generate QList structure which can be handled by the QConcurrent framework
115  QList<QPair<QList<FilterData>,QPair<int,RowVectorXd> > > timeData;
116 
117  //Only select channels specified in vecPicks
118  for(qint32 i = 0; i < vecPicks.cols(); ++i) {
119  timeData.append(QPair<QList<FilterData>,QPair<int,RowVectorXd> >(lFilterData,QPair<int,RowVectorXd>(vecPicks[i],matDataIn.row(vecPicks[i]))));
120  }
121 
122  //Do the concurrent filtering
123  if(!timeData.isEmpty()) {
124  QFuture<void> future = QtConcurrent::map(timeData,
125  doFilterPerChannelRTMSA);
126 
127  future.waitForFinished();
128 
129  //Do the overlap add method and store in matDataOut
130  int iFilteredNumberCols = timeData.at(0).second.second.cols();
131 
132  for(int r = 0; r < timeData.size(); r++) {
133  //Get the currently filtered data. This data has a delay of filterLength/2 in front and back.
134  RowVectorXd tempData = timeData.at(r).second.second;
135 
136  //Perform the actual overlap add by adding the last filterlength data to the newly filtered one
137  tempData.head(iOrder) += m_matOverlap.row(timeData.at(r).second.first);
138 
139  //Write the newly calulated filtered data to the filter data matrix. Keep in mind that the current block also effect last part of the last block (begin at dataIndex-iFilterDelay).
140  int start = 0;
141  matDataOut.row(timeData.at(r).second.first).segment(start,iFilteredNumberCols-iOrder) = tempData.head(iFilteredNumberCols-iOrder);
142 
143  //Refresh the m_matOverlap with the new calculated filtered data.
144  m_matOverlap.row(timeData.at(r).second.first) = timeData.at(r).second.second.tail(iOrder);
145  }
146  }
147 
148  if(matDataIn.cols() >= iOrder/2) {
149  m_matDelay = matDataIn.block(0, matDataIn.cols()-iOrder/2, matDataIn.rows(), iOrder/2);
150  } else {
151  qWarning() << "RtFilter::filterDataBlock - Half of filter length is larger than data size. Not filling m_matDelay for next step.";
152  }
153 
154  return matDataOut;
155 }

160 MatrixXd RtFilter::filterData(const MatrixXd& matDataIn,
161  FilterData::FilterType type,
162  double dCenterfreq,
163  double bandwidth,
164  double dTransition,
165  double dSFreq,
166  const RowVectorXi& vecPicks,
167  int iOrder,
168  qint32 iFftLength,
169  FilterData::DesignMethod designMethod)
170 {
171  // Check for size of data
172  if (matDataIn.cols()<iOrder){
173  qDebug() << QString("RtFilter::filterData - Filter length bigger then data length.");
174  }
175 
176  // Normalize cut off frequencies to nyquist
177  dCenterfreq = dCenterfreq/(dSFreq/2.0);
178  bandwidth = bandwidth/(dSFreq/2.0);
179  dTransition = dTransition/(dSFreq/2.0);
180 
181  // create output matrix with size of inputmatrix and temporal input matrix with size of pick
182  MatrixXd matDataOut = matDataIn;
183  MatrixXd sliceFiltered;
184  // create filter
185  FilterData filter = FilterData("rt_filter",
186  type,
187  iOrder,
188  dCenterfreq,
189  bandwidth,
190  dTransition,
191  dSFreq,
192  iFftLength,
193  designMethod);
194 
195  QList<FilterData> filterList;
196  filterList << filter;
197 
198  // slice input data into data junks with proper length for fft
199  int iSize = iFftLength-iOrder;
200 
201  if(matDataIn.cols() > iSize) {
202  int from = 0;
203  int numSlices = ceil(float(matDataIn.cols())/float(iSize)); //calculate number of data slices
204 
205  for (int i = 0; i<numSlices; i++) {
206  if(i == numSlices-1) {
207  //catch the last one that might be shorter then original size
208  iSize = matDataIn.cols() - (iSize * (numSlices -1));
209  }
210  sliceFiltered = filterDataBlock(matDataIn.block(0,from,matDataIn.rows(),iSize),
211  iOrder,
212  vecPicks,
213  filterList);
214  matDataOut.block(0,from,matDataIn.rows(),iSize) = sliceFiltered;
215  from += iSize;
216  }
217  } else {
218  matDataOut = filterDataBlock(matDataIn,
219  iOrder,
220  vecPicks,
221  filterList);
222  }
223  return matDataOut;
224 }
