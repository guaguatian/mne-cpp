366

369  {
370  m_pFiffInfo_Sensor = pRTMSA->info();
371

373  int arraySize = pRTMSA->getMultiArraySize();
374  int modulo = int(m_pFiffInfo_Sensor->sfreq*m_dSlidingWindowSize) % arraySize;

384  m_matTimeBetweenWindowsSensor.resize(rows, cols);
385 
386  m_matTimeBetweenWindowsStimSensor.resize(1,cols);

560 // cout<<"max: "<<max<<endl;

606 void BCI::BCIOnSensorLevel()
607 {
608  // Wait for fiff Info if not yet received - this is needed because we have to wait until the buffers are firstly initiated in the update functions
609  while(!m_pFiffInfo_Sensor)
610  msleep(10);
611 
612  // Start filling buffers with data from the inputs
613  m_bProcessData = true;
614 
615  // Sensor level: Fill matrices with data
616  if(m_bFillSensorWindowFirstTime) // Sensor level: Fill m_matSlidingWindowSensor with data for the first time
617  {
618  if(m_iTBWIndexSensor < m_matSlidingWindowSensor.cols())
619  {
620  //cout<<"About to pop matrix"<<endl;
621  MatrixXd t_mat = m_pBCIBuffer_Sensor->pop();
622  //cout<<"poped matrix"<<endl;
623 
624  // Get only the rows from the matrix which correspond with the selected features, namely electrodes on sensor level and destrieux clustered regions on source level
625  for(int i = 0; i < m_matSlidingWindowSensor.rows(); i++)
626  m_matSlidingWindowSensor.block(i, m_iTBWIndexSensor, 1, t_mat.cols()) = t_mat.block(m_mapElectrodePinningScheme[m_slChosenFeatureSensor.at(i)], 0, 1, t_mat.cols());
627 
628  m_matStimChannelSensor.block(0, m_iTBWIndexSensor, 1, t_mat.cols()) = t_mat.block(136, 0, 1, t_mat.cols());
629 
630  m_iTBWIndexSensor = m_iTBWIndexSensor + t_mat.cols();
631  }
632  else // m_matSlidingWindowSensor is full for the first time
633  {
634  m_iTBWIndexSensor = 0;
635  m_bFillSensorWindowFirstTime = false;
636  }
637  }
638  else // Sensor level: Fill m_matTimeBetweenWindowsSensor matrix until full -> Then -> recalculate m_matSlidingWindowSensor, calculate features and classify
639  {
640  if(m_iTBWIndexSensor < m_matTimeBetweenWindowsSensor.cols())
641  {
642  //cout<<"About to pop matrix"<<endl;
643  MatrixXd t_mat = m_pBCIBuffer_Sensor->pop();
644  //cout<<"poped matrix"<<endl;
645 
646  // Get only the rows from the matrix which correspond with the selected features, namely electrodes on sensor level and destrieux clustered regions on source level
647  for(int i = 0; i < m_matTimeBetweenWindowsSensor.rows(); i++)
648  m_matTimeBetweenWindowsSensor.block(i, m_iTBWIndexSensor, 1, t_mat.cols()) = t_mat.block(m_mapElectrodePinningScheme[m_slChosenFeatureSensor.at(i)], 0, 1, t_mat.cols());
649 
650  m_matTimeBetweenWindowsStimSensor.block(0, m_iTBWIndexSensor, 1, t_mat.cols()) = t_mat.block(136, 0, 1, t_mat.cols());
651 
652  m_iTBWIndexSensor = m_iTBWIndexSensor + t_mat.cols();
653  }
654  else // Recalculate m_matSlidingWindowSensor -> Calculate features, classify and store results
655  {
656  // ----1---- Recalculate m_matSlidingWindowSensor
657  //cout<<"----1----"<<endl;
