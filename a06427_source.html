147 bool MNE::read_events(QIODevice &p_IODevice, MatrixXi& eventlist)
148 {
149  //
150  // Open file
151  //
152  FiffStream::SPtr t_pStream(new FiffStream(&p_IODevice));
153 
154  if(!t_pStream->open()) {
155  return false;
156  }
157 
158  //
159  // Find the desired block
160  //
161  QList<FiffDirNode::SPtr> events = t_pStream->dirtree()->dir_tree_find(FIFFB_MNE_EVENTS);
162 
163  if (events.size() == 0)
164  {
165  printf("Could not find event data\n");
166  return false;
167  }
168 
169  qint32 k, nelem;
170  fiff_int_t kind, pos;
171  FiffTag::SPtr t_pTag;
172  quint32* serial_eventlist_uint = NULL;
173  qint32* serial_eventlist_int = NULL;
174 
175  for(k = 0; k < events[0]->nent(); ++k)
176  {
177  kind = events[0]->dir[k]->kind;
178  pos = events[0]->dir[k]->pos;
179  if (kind == FIFF_MNE_EVENT_LIST)
180  {
181  t_pStream->read_tag(t_pTag,pos);
182  if(t_pTag->type == FIFFT_UINT)
183  {
184  serial_eventlist_uint = t_pTag->toUnsignedInt();
185  nelem = t_pTag->size()/4;
186  }
187 
188  if(t_pTag->type == FIFFT_INT)
189  {
190  serial_eventlist_int = t_pTag->toInt();
191  nelem = t_pTag->size()/4;
192  }
193 
194  break;
195  }
196  }
197 
198  if(serial_eventlist_uint == NULL && serial_eventlist_int == NULL)
199  {
200  printf("Could not find any events\n");
201  return false;
202  }
203  else
204  {
205  eventlist.resize(nelem/3,3);
206  if(serial_eventlist_uint != NULL)
207  {
208  for(k = 0; k < nelem/3; ++k)
209  {
210  eventlist(k,0) = serial_eventlist_uint[k*3];
211  eventlist(k,1) = serial_eventlist_uint[k*3+1];
212  eventlist(k,2) = serial_eventlist_uint[k*3+2];
213  }
214  }
215 
216  if(serial_eventlist_int != NULL)
217  {
218  for(k = 0; k < nelem/3; ++k)
219  {
220  eventlist(k,0) = serial_eventlist_int[k*3];
221  eventlist(k,1) = serial_eventlist_int[k*3+1];
222  eventlist(k,2) = serial_eventlist_int[k*3+2];
223  }
224  }
225  }
226 
227  return true;
228 }
