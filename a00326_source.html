180 void AverageSceneItem::paintAveragePath(QPainter *painter)
181 {
182  if(m_lAverageData.size() == 0)
183  return;
184 
185  //get maximum range of respective channel type (range value in FiffChInfo does not seem to contain a reasonable value)
186  float dMaxValue = 1e-9f;
187 
188  switch(m_iChannelKind) {
189  case FIFFV_MEG_CH: {
190  if(m_iChannelUnit == FIFF_UNIT_T_M) { //gradiometers
191  dMaxValue = 1e-10f;
192  if(m_scaleMap.contains(FIFF_UNIT_T_M))
193  dMaxValue = m_scaleMap[FIFF_UNIT_T_M];
194  }
195  else if(m_iChannelUnit == FIFF_UNIT_T) //magnitometers
196  {
197  dMaxValue = 1e-11f;
198 
199  if(m_scaleMap.contains(FIFF_UNIT_T))
200  dMaxValue = m_scaleMap[FIFF_UNIT_T];
201  }
202  break;
203  }
204 
205  case FIFFV_REF_MEG_CH: { /*11/04/14 Added by Limin: MEG reference channel */
206  dMaxValue = 1e-11f;
207  if(m_scaleMap.contains(FIFF_UNIT_T))
208  dMaxValue = m_scaleMap[FIFF_UNIT_T];
209  break;
210  }
211  case FIFFV_EEG_CH: {
212  dMaxValue = 1e-4f;
213  if(m_scaleMap.contains(FIFFV_EEG_CH))
214  dMaxValue = m_scaleMap[FIFFV_EEG_CH];
215  break;
216  }
217  case FIFFV_EOG_CH: {
218  dMaxValue = 1e-3f;
219  if(m_scaleMap.contains(FIFFV_EOG_CH))
220  dMaxValue = m_scaleMap[FIFFV_EOG_CH];
221  break;
222  }
223  case FIFFV_STIM_CH: {
224  dMaxValue = 5;
225  if(m_scaleMap.contains(FIFFV_STIM_CH))
226  dMaxValue = m_scaleMap[FIFFV_STIM_CH];
227  break;
228  }
229  case FIFFV_MISC_CH: {
230  dMaxValue = 1e-3f;
231  if(m_scaleMap.contains(FIFFV_MISC_CH))
232  dMaxValue = m_scaleMap[FIFFV_MISC_CH];
233  break;
234  }
235  }
236 
237  //Plot averaged data
238  QRectF boundingRect = this->boundingRect();
239  double dScaleY = (boundingRect.height())/(2*dMaxValue);
240  QPointF qSamplePosition;
241 
242  //do for all currently stored evoked set data
243  for(int dataIndex = 0; dataIndex < m_lAverageData.size(); ++dataIndex) {
244  QString sAvrComment = m_lAverageData.at(dataIndex).first;
245 
246  if(m_qMapAverageActivation[sAvrComment]) {
247  //plot data from averaged data m_lAverageData with the calculated downsample factor
248  const double* averageData = m_lAverageData.at(dataIndex).second.first;
249  int totalCols = m_lAverageData.at(dataIndex).second.second;
250 
251  if(totalCols < m_iMaxWidth) {
252  m_rectBoundingRect = QRectF(-totalCols/2, -m_iMaxHeigth/2, totalCols, m_iMaxHeigth);
253  }
254 
255  //Calculate downsampling factor of averaged data in respect to the items width
256  int dsFactor;
257  totalCols / boundingRect.width()<1 ? dsFactor = 1 : dsFactor = totalCols / boundingRect.width();
258  if(dsFactor == 0) {
259  dsFactor = 1;
260  }
261 
262  //Create path
263  //float offset = (*(averageData+(0*m_iTotalNumberChannels)+m_iChannelNumber)); //choose offset to be the signal value at first sample
264  float offset = 0;
265  QPainterPath path = QPainterPath(QPointF(boundingRect.x(), boundingRect.y() + boundingRect.height()/2));
266  QPen pen;
267  pen.setStyle(Qt::SolidLine);
268  pen.setColor(Qt::yellow);
269 
270  if(m_qMapAverageColor.contains(sAvrComment)) {
271  pen.setColor(m_qMapAverageColor[sAvrComment]);
272  }
273 
274  pen.setWidthF(3);
275  painter->setPen(pen);
276 
277  for(int i = 0; i < totalCols && path.elementCount() <= boundingRect.width(); i += dsFactor) {
278  //evoked matrix is stored in column major
279  double val = ((*(averageData+(i*m_iTotalNumberChannels)+m_iChannelNumber))-offset) * dScaleY;
280 
281  //Cut plotting if six times bigger than m_iMaxHeigth
282  if(std::fabs(val) > 6*m_iMaxHeigth && m_bIsBad) {
283  qSamplePosition.setY(-(val/val) * m_iMaxHeigth); //(val/val) used to retrieve sign of val
284  qSamplePosition.setX(path.currentPosition().x()+1);
285  } else {
286  qSamplePosition.setY(-val);
287  qSamplePosition.setX(path.currentPosition().x()+1);
288  }
289 
290  path.lineTo(qSamplePosition);
291  }
292 
293  painter->drawPath(path);
294  }
295  }
