245 void NeuronalConnectivity::updateSource(SCMEASLIB::Measurement::SPtr pMeasurement)
246 {
247  QSharedPointer<RealTimeSourceEstimate> pRTSE = pMeasurement.dynamicCast<RealTimeSourceEstimate>();
248 
249  if(pRTSE) {
250  //Fiff information
251  if(!m_pFiffInfo) {
252  m_pFiffInfo = pRTSE->getFiffInfo();
253 
254  m_connectivitySettings.setSamplingFrequency(m_pFiffInfo->sfreq);
255 
256  //Init output - Unocmment this if you also uncommented the m_pRTCEOutput in the constructor above
257  m_pRTCEOutput->data()->setAnnotSet(pRTSE->getAnnotSet());
258  m_pRTCEOutput->data()->setSurfSet(pRTSE->getSurfSet());
259  m_pRTCEOutput->data()->setFwdSolution(pRTSE->getFwdSolution());
260  m_pRTCEOutput->data()->setFiffInfo(m_pFiffInfo);
261 
262  // Generate network nodes
263  m_connectivitySettings.setNodePositions(*pRTSE->getFwdSolution(), *pRTSE->getSurfSet());
264  }
265 
266  for(qint32 i = 0; i < pRTSE->getValue().size(); ++i) {
267  // Find out how many samples were used for pre stimulus
268  int iZeroIdx = 0;
269  for(int j = 0; j < pRTSE->getValue()[i]->times.cols(); ++j) {
270  if(pRTSE->getValue()[i]->times(j) >= 0) {
271  iZeroIdx = j;
272  //iZeroIdx = j + m_pFiffInfo->sfreq * 0.01; //Cut stimulus artifact, e.g. for median nerve stimulation.
273  break;
274  }
275  }
276 
277  m_iBlockSize = pRTSE->getValue().first()->data.cols() - iZeroIdx;
278 
279  // Check row and colum integrity and restart if necessary
280  if(m_connectivitySettings.size() != 0) {
281  if(m_iBlockSize != m_connectivitySettings.at(0).matData.cols()) {
282  m_connectivitySettings.clearAllData();
283  m_pRtConnectivity->restart();
284  }
285  }
286 
287  m_connectivitySettings.append(pRTSE->getValue()[i]->data.block(0,
288  iZeroIdx,
289  pRTSE->getValue()[i]->data.rows(),
290  pRTSE->getValue()[i]->data.cols() - iZeroIdx));
291  }
292 
293  //Pop data from buffer
294  if(m_connectivitySettings.size() >= m_iNumberAverages) {
295  m_pRtConnectivity->restart();
296  m_connectivitySettings.removeFirst(m_connectivitySettings.size()-m_iNumberAverages);
297  }
298 
299  m_timer.restart();
300  m_pRtConnectivity->append(m_connectivitySettings);
301  }

464 void NeuronalConnectivity::generateNodeVertices()
465 {
466  if(!m_pFiffInfo) {
467  qDebug() << "NeuronalConnectivity::generateNodeVertices - FiffInfo is Null. Returning.";
468  return;
469  }
470 
471  QString sCoilType = "grad";
472  QString sChType = "meg";
473 
474  QStringList exclude;
475  exclude << m_pFiffInfo->bads << m_pFiffInfo->ch_names.filter("EOG");
476 
477  if(sChType.contains("EEG", Qt::CaseInsensitive)) {
478  m_vecPicks = m_pFiffInfo->pick_types(false,true,false,QStringList(),exclude);
479  } else if(sCoilType.contains("grad", Qt::CaseInsensitive)) {
480  // Only pick every second gradiometer which are not marked as bad.
481  RowVectorXi picksTmp = m_pFiffInfo->pick_types(QString("grad"),false,false);
482  m_vecPicks.resize(0);
483 
484  for(int i = 0; i < picksTmp.cols()-1; i+=2) {
485  if(!m_pFiffInfo->bads.contains(m_pFiffInfo->ch_names.at(picksTmp(i)))) {
486  m_vecPicks.conservativeResize(m_vecPicks.cols()+1);
487  m_vecPicks(m_vecPicks.cols()-1) = picksTmp(i);
488  }
489  }
490  } else if (sCoilType.contains("mag", Qt::CaseInsensitive)) {
491  m_vecPicks = m_pFiffInfo->pick_types(QString("mag"),false,false,QStringList(),exclude);
492  }
493 
494  // Set sampling frequency so that the spectrum resolution is updated
495  m_connectivitySettings.setSamplingFrequency(m_pFiffInfo->sfreq);
496 
497  //Set node 3D positions to connectivity settings
498  m_connectivitySettings.setNodePositions(*m_pFiffInfo, m_vecPicks);
499  m_connectivitySettings.clearAllData();
500 }

505 void NeuronalConnectivity::run()
506 {

508  while(!m_pFiffInfo) {
509  msleep(10);
510  }
511 
512  int skip_count = 0;
513 
514  while(m_bIsRunning) {
515  //Do processing after skip count has reached limit
516  if((skip_count % m_iDownSample) == 0) {
517  //QMutexLocker locker(&m_mutex);
518  //Do connectivity estimation here
519  m_currentConnectivityResult = m_pCircularNetworkBuffer->pop();
520 
521  //Send the data to the connected plugins and the online display
522  if(!m_currentConnectivityResult.isEmpty()) {
523  //qDebug()<<"NeuronalConnectivity::run - Total time"<<m_timer.elapsed();
524  m_currentConnectivityResult.setFrequencyRange(m_fFreqBandLow, m_fFreqBandHigh);
525  m_currentConnectivityResult.normalize();
526 
527  m_pRTCEOutput->data()->setValue(m_currentConnectivityResult);
528  } else {
529  qDebug()<<"NeuronalConnectivity::run - Network is empty";
530  }
531 
532  }
533 
534  ++skip_count;
535  }
