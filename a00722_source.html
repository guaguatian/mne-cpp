252  QList<qint32> listSourcesVertNoLabel = m_lInterpolationData.mapLabelIdSources.keys(label.label_id);
253 
254  for(int j = 0; j < label.vertices.rows(); ++j) {

256  int colIdx = m_lInterpolationData.vertNos.indexOf(listSourcesVertNoLabel.at(k));
257  m_pMatAnnotationMat->coeffRef(label.vertices(j),colIdx) = 1.0f/listSourcesVertNoLabel.size();
258  }
