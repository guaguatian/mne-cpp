227 QWidget* LSLAdapter::setupWidget()
228 {
229  // apparently the widget may get deleted during runtime,
230  // so we have to return a new one everytime this method is called
231  LSLAdapterSetup* temp = new LSLAdapterSetup(m_iOutputBlockSize);
232  connect(temp, &LSLAdapterSetup::refreshAvailableStreams, this, &LSLAdapter::onRefreshAvailableStreams);
233  connect(temp, &LSLAdapterSetup::streamSelectionChanged, this, &LSLAdapter::onStreamSelectionChanged);
234  connect(temp, &LSLAdapterSetup::blockSizeChanged, this, &LSLAdapter::onBlockSizeChanged);
235  connect(this, &LSLAdapter::updatedAvailableLSLStreams, temp, &LSLAdapterSetup::onLSLScanResults);
236 
237  // check if we have some information about previously available lsl streams:
238  if(m_vAvailableStreams.isEmpty() == false && m_bHasValidStream) {
239  // let the widget display potentially outdated info, until the background thread for stream scanning will return
240  temp->onLSLScanResults(m_vAvailableStreams, m_currentStream);
241  }
242 
243  // try to start background scan for available LSL streams
244  onRefreshAvailableStreams();
245 
246  return temp;

272 void LSLAdapter::onLSLStreamScanReady()
273 {
274  // save result streams
275  m_vAvailableStreams = m_updateStreamsFutureWatcher.result();
276 
277  // check whether any streams are available
278  if(m_vAvailableStreams.size() == 0) {
279  m_bHasValidStream = false;
280  // overwrite current stream with default constructor, this will also result in correct UI display
281  m_currentStream = lsl::stream_info();
282  }
283  else {
284  // check whether we had a valid stream, and if its still amongst the available ones
285  if(m_bHasValidStream) {
286  if(contains(m_vAvailableStreams, m_currentStream) == false) {
287  qDebug() << "[LSLAdapter] Old stream no longer available, switching to first available stream";
288  m_currentStream = m_vAvailableStreams[0];
289  m_bHasValidStream = true;
290  }
291  // else-case: no need to change anything
292  } else {
293  // simply take first stream
294  m_currentStream = m_vAvailableStreams[0];
295  m_bHasValidStream = true;
296  }
297  }
298 
299  // tell UI
300  emit updatedAvailableLSLStreams(m_vAvailableStreams, m_currentStream);
301 }

335 void LSLAdapter::prepareFiffInfo(const lsl::stream_info &stream)
336 {
337  // parse fiff info from lsl stream info
338  QString type = QString::fromStdString(stream.type()).toUpper();
339  if(type == "EEG") {
340  // copy basic info into intended member fields
341  m_iNumberChannels = stream.channel_count();
342  m_fSamplingFrequency = static_cast<float>(stream.nominal_srate());
343 
344  // clear old fiff info data
345  m_pFiffInfo->clear();
346  // set number of channels, sampling frequency and high/-lowpass
347  m_pFiffInfo->nchan = m_iNumberChannels;
348  m_pFiffInfo->sfreq = m_fSamplingFrequency;
349 
350  // set up the channel info
351  QStringList QSLChNames;
352  m_pFiffInfo->chs.clear();
353 
354  for(int i = 0; i < m_pFiffInfo->nchan; ++i) {
355  // create information for each channel
356  QString sChType;
357  FiffChInfo fChInfo;
358 
359  // set channel name
360  sChType = QString("EEG ");
361  if(i < 10) {
362  sChType.append("00");
363  }
364 
365  if(i >= 10 && i < 100) {
366  sChType.append("0");
367  }
368 
369  fChInfo.ch_name = sChType.append(sChType.number(i));
370 
371  // set channel type
372  fChInfo.kind = FIFFV_EEG_CH;
373 
374  // set logno
375  fChInfo.logNo = i;
376 
377  // set coord frame
378  fChInfo.coord_frame = FIFFV_COORD_HEAD;
379 
380  // set unit
381  fChInfo.unit = FIFF_UNIT_V;
382  fChInfo.unit_mul = 0;
383 
384  // set EEG electrode location - Convert from mm to m
385  fChInfo.eeg_loc(0,0) = 0;
386  fChInfo.eeg_loc(1,0) = 0;
387  fChInfo.eeg_loc(2,0) = 0;

396  fChInfo.chpos.r0(1) = 0;
397  fChInfo.chpos.r0(2) = 0;
