487 void FilterDesignView::filterParametersChanged()
488 {
489  //User defined filter parameters
490  double lowpassHz = ui->m_doubleSpinBox_lowpass->value();
491  double highpassHz = ui->m_doubleSpinBox_highpass->value();
492 
493  double trans_width = ui->m_doubleSpinBox_transitionband->value();
494 
495  double bw = highpassHz-lowpassHz;
496  double center = lowpassHz+bw/2;
497 
498  double samplingFrequency = m_dSFreq <= 0 ? 600 : m_dSFreq;
499  double nyquistFrequency = samplingFrequency/2;
500 
501  //Calculate the needed fft length
502  m_iFilterTaps = ui->m_spinBox_filterTaps->value();
503  if(ui->m_spinBox_filterTaps->value()%2 != 0)
504  m_iFilterTaps--;
505 
506  int fftLength = m_iWindowSize + ui->m_spinBox_filterTaps->value() * 4; // *2 to take into account the overlap in front and back after the convolution. Another *2 to take into account the appended and prepended data.
507  int exp = ceil(MNEMath::log2(fftLength));
508  fftLength = pow(2, exp) <512 ? 512 : pow(2, exp);
509 
510 // qDebug() <<"fftLength: "<<fftLength;
511 // qDebug()<<"m_iWindowSize: "<<m_iWindowSize;
512 // qDebug()<<"m_iWindowSize + ui->m_spinBox_filterTaps->value() * 4: "<<m_iWindowSize + ui->m_spinBox_filterTaps->value() * 4;
513 
514  //set maximum and minimum for cut off frequency spin boxes
515  ui->m_doubleSpinBox_highpass->setMaximum(nyquistFrequency);
516  ui->m_doubleSpinBox_lowpass->setMaximum(nyquistFrequency);
517  ui->m_doubleSpinBox_highpass->setMinimum(0);
518  ui->m_doubleSpinBox_lowpass->setMinimum(0);
519 
520  if(ui->m_comboBox_filterType->currentText() == "Bandpass") {
521  if((ui->m_doubleSpinBox_highpass->value() < ui->m_doubleSpinBox_lowpass->value())) {
522  ui->m_doubleSpinBox_highpass->setValue(ui->m_doubleSpinBox_lowpass->value());
523  }
524  ui->m_doubleSpinBox_highpass->setMinimum(ui->m_doubleSpinBox_lowpass->value());
525  ui->m_doubleSpinBox_lowpass->setMaximum(ui->m_doubleSpinBox_highpass->value());
526  }
527 
528  //set filter design method
529  FilterData::DesignMethod dMethod = FilterData::Tschebyscheff;
530  if(ui->m_comboBox_designMethod->currentText() == "Tschebyscheff") {
531  dMethod = FilterData::Tschebyscheff;
532  }
533 
534  if(ui->m_comboBox_designMethod->currentText() == "Cosine") {
535  dMethod = FilterData::Cosine;
536  }
537 
538  //Generate filters
539  //Note: Always use "User Design" as filter name for user designed filters, which are stored in the model. This needs to be done because there only should be one filter in this model which holds the user designed filter.
540  //Otherwise everytime a filter is designed a new filter would be added to this model -> too much storage consumption.
541  if(ui->m_comboBox_filterType->currentText() == "Lowpass") {
542  m_filterData = FilterData("User Design",
543  FilterData::LPF,
544  m_iFilterTaps,
545  lowpassHz/nyquistFrequency,
546  0.2,
547  (double)trans_width/nyquistFrequency,
548  samplingFrequency,
549  fftLength,
550  dMethod);
551  }
552 
553  if(ui->m_comboBox_filterType->currentText() == "Highpass") {
554  m_filterData = FilterData("User Design",
555  FilterData::HPF,
556  m_iFilterTaps,
557  highpassHz/nyquistFrequency,
558  0.2,
559  (double)trans_width/nyquistFrequency,
560  samplingFrequency,
561  fftLength,
562  dMethod);
563  }
564 
565  if(ui->m_comboBox_filterType->currentText() == "Bandpass") {
566  m_filterData = FilterData("User Design",
567  FilterData::BPF,
568  m_iFilterTaps,
569  (double)center/nyquistFrequency,
570  (double)bw/nyquistFrequency,
571  (double)trans_width/nyquistFrequency,
572  samplingFrequency,
573  fftLength,
574  dMethod);
575  }
576 
577  emit filterChanged(m_filterData);
578 
579  //update filter plot
580  updateFilterPlot();
581 }
