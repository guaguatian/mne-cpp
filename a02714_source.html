360 bool GUSBAmpDriver::getSampleMatrixValue(MatrixXf& sampleMatrix)
361 {
362  sampleMatrix.setZero(); // Clear matrix - set all elements to zero
363 
364  for(int queueIndex=0; queueIndex<m_QUEUE_SIZE; queueIndex++)
365  {
366  //receive data from each device
367  for (int deviceIndex = 0; deviceIndex < m_numDevices; deviceIndex++)
368  {
369  HANDLE hDevice = m_callSequenceHandles[deviceIndex];
370 
371  //wait for notification from the system telling that new data is available
372  if (WaitForSingleObject(m_overlapped[deviceIndex][queueIndex].hEvent, 1000) == WAIT_TIMEOUT)
373  {
374  //throw string("Error on data transfer: timeout occurred.");
375  cout << "Error on data transfer: timeout occurred." << "\n";
376  return 0;
377  }
378 
379  //get number of received bytes...
380  GetOverlappedResult(hDevice, &m_overlapped[deviceIndex][queueIndex], &m_numBytesReceived, false);
381 
382  //...and check if we lost something (number of received bytes must be equal to the previously allocated buffer size)
383  if (m_numBytesReceived != m_bufferSizeBytes)
384  {
385  //throw string("Error on data transfer: samples lost.");
386  cout << "Error on data transfer: samples lost." << "\n";
387  return 0;
388  }
389  }
390 
391  //store received data from each device in the correct order (that is scan-wise, where one scan includes all channels of all devices) ignoring the header
392  //Data is aligned as follows: element at position destBuffer(scanIndex * (m_chNumberOfChannels + m_bTrigger) + channelIndex) * sizeof(float) + HEADER_SIZE is sample of channel channelIndex (zero-based) of the scan with zero-based scanIndex.
393  //channelIndex ranges from 0..numDevices*numChannelsPerDevices where numDevices equals the number of recorded devices and numChannelsPerDevice the number of channels from each of those devices.
394  //It is assumed that all devices provide the same number of channels.
395  for (int scanIndex = 0; scanIndex < m_iNumberOfScans; scanIndex++)
396  {
397  for (int deviceIndex = 0; deviceIndex < m_numDevices; deviceIndex++)
398  {
399  for(int channelIndex = 0; channelIndex<m_chNumberOfChannels; channelIndex++)
400  {
401  BYTE ByteValue[sizeof(float)];
402  float FloatValue;
403 
404  for(int i=0;i<sizeof(float);i++)
405  {
406  ByteValue[i] = m_buffers[deviceIndex][queueIndex][(scanIndex * (m_chNumberOfChannels + m_bTrigger) + channelIndex) * sizeof(float) + HEADER_SIZE + i];
407  }
408  memcpy(&FloatValue, &ByteValue, sizeof(float));
409 
410  //store float-value to Matrix
411  sampleMatrix(channelIndex + deviceIndex*int(m_chNumberOfChannels), scanIndex + queueIndex * m_iNumberOfScans) = FloatValue;
412  }
413  }
414  }
415 
416  //add new GetData call to the queue replacing the currently received one
417  for (int deviceIndex = 0; deviceIndex < m_numDevices; deviceIndex++)
418  if (!GT_GetData(m_callSequenceHandles[deviceIndex], m_buffers[deviceIndex][queueIndex], m_bufferSizeBytes, &m_overlapped[deviceIndex][queueIndex]))
419  {
420  cout << "\tError on GT_GetData.\n";
421  return 0;
422  }
423  }
424  return true;
425 }

464  m_callSequenceSerials.push_back(m_vpSerials[0]);
