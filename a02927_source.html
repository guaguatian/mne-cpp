219 void RtHpi::update(SCMEASLIB::Measurement::SPtr pMeasurement)
220 {
221  QSharedPointer<RealTimeMultiSampleArray> pRTMSA = pMeasurement.dynamicCast<RealTimeMultiSampleArray>();
222

225  m_qMutex.lock();
226  //Check if buffer initialized
227  if(!m_pRtHpiBuffer)
228  m_pRtHpiBuffer = CircularMatrixBuffer<double>::SPtr(new CircularMatrixBuffer<double>(8, pRTMSA->getNumChannels(), pRTMSA->getMultiSampleArray()[0].cols()));
229 
230  //Fiff information
231  if(!m_pFiffInfo)
232  {
233  m_pFiffInfo = pRTMSA->info();
234  emit fiffInfoAvailable();
235  }
236 
237  m_qMutex.unlock();
238  if(m_bProcessData)

240  MatrixXd t_mat;

246  }

255 void RtHpi::run()

257  //
258  // Read Fiff Info and wait if absent
259  //

270 
271  m_qMutex.lock();

273  m_qMutex.unlock();
274 
275  qDebug()<<"+++++++++++ Start Real time HPI thread +++++++++++++++++++";
276

278
