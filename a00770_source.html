280 void HpiView::updateProjections()
281 {
282  m_matProjectors = Eigen::MatrixXd::Identity(m_pFiffInfo->chs.size(), m_pFiffInfo->chs.size());
283  Eigen::MatrixXd matComp = Eigen::MatrixXd::Identity(m_pFiffInfo->chs.size(), m_pFiffInfo->chs.size());
284 
285  if(m_bUseSSP) {
286  // Use SSP + SGM + calibration
287  //Do a copy here because we are going to change the activity flags of the SSP's
288  FiffInfo infoTemp = *(m_pFiffInfo.data());
289 
290  //Turn on all SSP
291  for(int i = 0; i < infoTemp.projs.size(); ++i) {
292  infoTemp.projs[i].active = true;
293  }
294 
295  //Create the projector for all SSP's on
296  infoTemp.make_projector(m_matProjectors);
297  //set columns of matrix to zero depending on bad channels indexes
298  for(qint32 j = 0; j < infoTemp.bads.size(); ++j) {
299  m_matProjectors.col(infoTemp.ch_names.indexOf(infoTemp.bads.at(j))).setZero();
300  }
301  }
302 
303  if(m_bUseComp) {
304  // Setup Comps
305  FiffCtfComp newComp;
306  //Do this always from 0 since we always read new raw data, we never actually perform a multiplication on already existing data
307  if(m_pFiffInfo->make_compensator(0, 101, newComp)) {
308  matComp = newComp.data->data;
309  }
310  }
311 
312  m_matCompProjectors = m_matProjectors * matComp;
313 
314  m_pRtHPI->setProjectionMatrix(m_matProjectors);
315 }

332 QList<FiffDigPoint> HpiView::readPolhemusDig(const QString& fileName)
333 {
334  QFile t_fileDig(fileName);
335  FiffDigPointSet t_digSet(t_fileDig);
336 
337  QList<FiffDigPoint> lDigPoints;
338 
339  qint16 numHPI = 0;
340  qint16 numDig = 0;
341  qint16 numFiducials = 0;
342  qint16 numEEG = 0;
343 
344  for(int i = 0; i < t_digSet.size(); ++i) {
345  lDigPoints.append(t_digSet[i]);
346 
347  switch(t_digSet[i].kind)
348  {
349  case FIFFV_POINT_HPI:
350  numHPI++;
351  break;
352 
353  case FIFFV_POINT_EXTRA:
354  numDig++;
355  break;
356 
357  case FIFFV_POINT_CARDINAL:
358  numFiducials++;
359  break;
360 
361  case FIFFV_POINT_EEG:
362  numEEG++;
363  break;
364  }
365  }
366 
367  //Add all digitizer but additional points to the 3D view
368  FiffDigPointSet t_digSetWithoutAdditional = t_digSet.pickTypes(QList<int>()<<FIFFV_POINT_HPI<<FIFFV_POINT_CARDINAL<<FIFFV_POINT_EEG);
369  m_pTrackedDigitizer = m_pData3DModel->addDigitizerData("Head", "Tracked", t_digSetWithoutAdditional);
370  //m_pData3DModel->addDigitizerData("Head", "Tracked", t_digSetWithoutAdditional);
371 
372  //Set loaded number of digitizers
373  ui->m_label_numberLoadedCoils->setNum(numHPI);
374  ui->m_label_numberLoadedDigitizers->setNum(numDig);
375  ui->m_label_numberLoadedFiducials->setNum(numFiducials);
376  ui->m_label_numberLoadedEEG->setNum(numEEG);
377 
378  //Hide/show frequencies and errors based on the number of coils
379  if(numHPI == 3) {
380  ui->m_label_gofCoil4->hide();
381  ui->m_label_gofCoil4Description->hide();
382  ui->m_label_freqCoil4->hide();
383  ui->m_spinBox_freqCoil4->hide();
384 
385  m_vCoilFreqs.clear();
386  m_vCoilFreqs << 155 << 165 << 190;
387  } else {
388  ui->m_label_gofCoil4->show();
389  ui->m_label_gofCoil4Description->show();
390  ui->m_label_freqCoil4->show();
391  ui->m_spinBox_freqCoil4->show();
392 
393  m_vCoilFreqs.clear();
394  m_vCoilFreqs << 155 << 165 << 190 << 220;
395  }
396 
397  //alignFiducials(fileName);
398 
399  return lDigPoints;
