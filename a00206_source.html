195 
196  if (m_pSeries->count() == 0) //protect integrity of the histogram widget in case series contain no data values

199  }
200 
201 // the condition below tests the threshold values given and ensures that all three must be within minAxisX and maxAxisX otherwise they will be given either minAxisX or maxAxisX value
202  else if (correctedVectorThreshold.x() < m_dMinAxisX || correctedVectorThreshold.y() < m_dMinAxisX || correctedVectorThreshold.z() < m_dMinAxisX || correctedVectorThreshold.x() > m_dMaxAxisX || correctedVectorThreshold.y() > m_dMaxAxisX || correctedVectorThreshold.z() > m_dMaxAxisX)

308 void Spline::setColorMap(const QString& colorMap)
309 {
310  m_colorMap = colorMap;
311  double leftThresholdValue = (m_pLeftThreshold->at(0).x())/ m_dMaxAxisX;
312  double middleThresholdValue = (m_pMiddleThreshold->at(0).x())/ m_dMaxAxisX;
313  double rightThresholdValue = (m_pRightThreshold->at(0).x())/ m_dMaxAxisX;
314  int stepsNumber = 25;
315  double stepsSizeLeftMiddle = (middleThresholdValue - leftThresholdValue) / stepsNumber;
316  double stepsSizeMiddleRight = (rightThresholdValue - middleThresholdValue) / stepsNumber;
317  QLinearGradient plotAreaGradient;
318  plotAreaGradient.setStart(QPointF(0, 0));
319  plotAreaGradient.setFinalStop(QPointF(1, 0));
320 
321  plotAreaGradient.setColorAt(leftThresholdValue, ColorMap::valueToColor(0.0, colorMap));
322 
323  for (int i = 1; i < stepsNumber; ++i)
324  {
325  plotAreaGradient.setColorAt(leftThresholdValue + (stepsSizeLeftMiddle * i), ColorMap::valueToColor((double)i * (0.5 / (double)stepsNumber), colorMap));
326  plotAreaGradient.setColorAt(middleThresholdValue + (stepsSizeMiddleRight * i), ColorMap::valueToColor((double)0.5 + (i * (0.5 / (double)stepsNumber)), colorMap));
327  }
328  plotAreaGradient.setColorAt(rightThresholdValue, ColorMap::valueToColor(1.0, colorMap));
329 
330  // Customize plot area background
331  plotAreaGradient.setCoordinateMode(QGradient::ObjectBoundingMode);
332  m_pChart->setPlotAreaBackgroundBrush(plotAreaGradient);
333  m_pChart->setPlotAreaBackgroundVisible(true);
334 }

377  }

382  {
383  exponent = -(std::abs(m_vecResultExponentValues[0]));
384  }
385  else if (m_vecResultExponentValues[0] > 0)
386  {
387  exponent = std::abs(m_vecResultExponentValues[0]);

392  }
393  }
394  else

396  qDebug() << "Spline::correctionDisplayTrueValue error.";
